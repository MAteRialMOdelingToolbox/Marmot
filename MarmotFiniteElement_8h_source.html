53 template < int nDim, int nNodes >
54 Eigen::Matrix< double, nDim, nDim * nNodes > NB( const Eigen::Matrix< double, 1, nNodes >& N )
55 {
56 // Alternative Templated version of Interpolation operator NBold;
57 Eigen::Matrix< double, nDim, nDim* nNodes > N_ = Eigen::Matrix< double, nDim, nDim * nNodes >::Zero();
58 for ( int i = 0; i < nNodes; i++ ) {
59 for ( int j = 0; j < nDim; j++ ) {
60 N_( j, nDim * i + j ) = N( i );
61 }
62 }
63 return N_;
64 }

69 template < int nDim, int nNodes >
70 Eigen::Matrix< double, nDim, nDim > Jacobian( const Eigen::Matrix< double, nDim, nNodes >& dNdXi,
71 const Eigen::Matrix< double, nDim * nNodes, 1 >& coordinates )
72 {
73 // Alternative Templated version of Jacobian for compile time known
74 // sizes
75 Eigen::Matrix< double, nDim, nDim > J_ = Eigen::Matrix< double, nDim, nDim >::Zero();
76 for ( int i = 0; i < nDim; i++ ) // loop over global dimensions
77 for ( int j = 0; j < nDim; j++ ) // loop over local dimensions
78 for ( int k = 0; k < nNodes; k++ ) // Loop over nodes
79 J_( i, j ) += dNdXi( j, k ) * coordinates( i + k * nDim );
80 return J_;
81 }

111 template < int nNodes >
112 Eigen::Matrix< double, voigtSize, nNodes * nDim > B( const Eigen::Matrix< double, nDim, nNodes >& dNdX )
113 {
114
115 Eigen::Matrix< double, voigtSize, nNodes* nDim > B_ = Eigen::Matrix< double, voigtSize, nNodes * nDim >::Zero();
116 for ( int i = 0; i < nNodes; i++ ) {
117 B_( 0, 2 * i ) = dNdX( 0, i );
118 B_( 1, 2 * i + 1 ) = dNdX( 1, i );
119 B_( 2, 2 * i ) = dNdX( 1, i );
120 B_( 2, 2 * i + 1 ) = dNdX( 0, i );
121 }
122 return B_;
123 }

125 template < int nNodes >
126 Eigen::Matrix< double, voigtSize, nNodes * nDim > BGreen( const Eigen::Matrix< double, nDim, nNodes >& dNdX,
127 const Eigen::Matrix2d& F )
128 {
129 // Green-Lagrange Strain Operator for given dNdX and Deformationgradient F
130 // Belytschko et. al pp. 213
131 Eigen::Matrix< double, voigtSize, nNodes* nDim > B_ = Eigen::Matrix< double, voigtSize, nNodes * nDim >::Zero();
132 for ( int i = 0; i < nNodes; i++ ) {
133 B_( 0, 2 * i ) = dNdX( 0, i ) * F( 0, 0 );
134 B_( 0, 2 * i + 1 ) = dNdX( 0, i ) * F( 1, 0 );
135 B_( 1, 2 * i ) = dNdX( 1, i ) * F( 0, 1 );
136 B_( 1, 2 * i + 1 ) = dNdX( 1, i ) * F( 1, 1 );
137 B_( 2, 2 * i ) = dNdX( 0, i ) * F( 0, 1 ) + dNdX( 1, i ) * F( 0, 0 );
138 B_( 2, 2 * i + 1 ) = dNdX( 0, i ) * F( 1, 1 ) + dNdX( 1, i ) * F( 1, 0 );
139 }
140 return B_;
141 }

175 template < int nNodes >
176 Eigen::Matrix< double, voigtSize, nNodes * nDim > B( const Eigen::Matrix< double, nDim, nNodes >& dNdX )
177 {
178 // ABAQUS like notation of strain: ( 11, 22, 33, 12, 13, 23 )
179 // _ _
180 // | dN/dx1 0 0 |
181 // | 0 dN/dx2 0 |
182 // | 0 0 dN/dx3 |
183 // | dN/dx2 dN/dx1 0 |
184 // | dN/dx3 0 dN/dx1 |
185 // |_ 0 dN/dx3 dN/dx2 _|
186
187 Eigen::Matrix< double, voigtSize, nNodes* nDim > B_ = Eigen::Matrix< double, voigtSize, nNodes * nDim >::Zero();
188
189 for ( int i = 0; i < nNodes; i++ ) {
190 B_( 0, nDim * i ) = dNdX( 0, i );
191 B_( 1, nDim * i + 1 ) = dNdX( 1, i );
192 B_( 2, nDim * i + 2 ) = dNdX( 2, i );
193 B_( 3, nDim * i + 0 ) = dNdX( 1, i );
194 B_( 3, nDim * i + 1 ) = dNdX( 0, i );
195 B_( 4, nDim * i + 0 ) = dNdX( 2, i );
196 B_( 4, nDim * i + 2 ) = dNdX( 0, i );
197 B_( 5, nDim * i + 1 ) = dNdX( 2, i );
198 B_( 5, nDim * i + 2 ) = dNdX( 1, i );
199 }
200
201 return B_;
202 }

204 template < int nNodes >
205 Eigen::Matrix< double, voigtSize, nNodes * nDim > BGreen( const Eigen::Matrix< double, nDim, nNodes >& dNdX,
206 const Eigen::Matrix3d& F )
207 {
208 // Green-Lagrange Strain Operator for given dNdX and Deformationgradient F
209 // Belytschko et. al pp. 213
210
211 Eigen::Matrix< double, voigtSize, nNodes* nDim > B_ = Eigen::Matrix< double, voigtSize, nNodes * nDim >::Zero();
212 for ( int i = 0; i < nNodes; i++ ) {
213 B_( 0, nDim * i ) = dNdX( 0, i ) * F( 0, 0 );
214 B_( 0, nDim * i + 1 ) = dNdX( 0, i ) * F( 1, 0 );
215 B_( 0, nDim * i + 2 ) = dNdX( 0, i ) * F( 2, 0 );
216
217 B_( 1, nDim * i ) = dNdX( 1, i ) * F( 0, 1 );
218 B_( 1, nDim * i + 1 ) = dNdX( 1, i ) * F( 1, 1 );
219 B_( 1, nDim * i + 2 ) = dNdX( 1, i ) * F( 2, 1 );
220
221 B_( 2, nDim * i ) = dNdX( 2, i ) * F( 0, 2 );
222 B_( 2, nDim * i + 1 ) = dNdX( 2, i ) * F( 1, 2 );
223 B_( 2, nDim * i + 2 ) = dNdX( 2, i ) * F( 2, 2 );
224
225 B_( 3, nDim * i ) = dNdX( 0, i ) * F( 0, 1 ) + dNdX( 1, i ) * F( 0, 0 );
226 B_( 3, nDim * i + 1 ) = dNdX( 0, i ) * F( 1, 1 ) + dNdX( 1, i ) * F( 1, 0 );
227 B_( 3, nDim * i + 2 ) = dNdX( 0, i ) * F( 2, 1 ) + dNdX( 1, i ) * F( 2, 0 );
228
229 B_( 4, nDim * i ) = dNdX( 0, i ) * F( 0, 2 ) + dNdX( 2, i ) * F( 0, 0 );
230 B_( 4, nDim * i + 1 ) = dNdX( 0, i ) * F( 1, 2 ) + dNdX( 2, i ) * F( 1, 0 );
231 B_( 4, nDim * i + 2 ) = dNdX( 0, i ) * F( 2, 2 ) + dNdX( 2, i ) * F( 2, 0 );
232
233 B_( 5, nDim * i ) = dNdX( 2, i ) * F( 0, 1 ) + dNdX( 1, i ) * F( 0, 2 );
234 B_( 5, nDim * i + 1 ) = dNdX( 2, i ) * F( 1, 1 ) + dNdX( 1, i ) * F( 1, 2 );
235 B_( 5, nDim * i + 2 ) = dNdX( 2, i ) * F( 2, 1 ) + dNdX( 1, i ) * F( 2, 2 );
236 }
237 return B_;
