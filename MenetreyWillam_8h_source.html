146 template < typename T >
147 static T polarRadius( const T& theta, const double& e )
148 {
149 // computes the deviatoric shape roundness for a given eccentricity (e) at a certain
150 // position (theta) the numerator and denominator are stored for performance reasons, as
151 // they are also needed for the derivative dRdTheta
152 if ( e >= 1.0 )
153 return 1;
154
155 const double e2 = e * e;
156 const T cosTheta = cos( theta );
157 const T cos2Theta = cosTheta * cosTheta;
158
159 const T numerator = ( 4 * ( 1 - e2 ) * cos2Theta + ( 2 * e - 1 ) * ( 2 * e - 1 ) );
160 const T denominator = ( 2 * ( 1 - e2 ) * cosTheta +
161 ( 2 * e - 1 ) * sqrt( 4 * ( 1 - e2 ) * cos2Theta + 5 * e2 - 4 * e ) );
162
163 return numerator / denominator;
164 }

181 template < typename T >
182 static std::pair< T, T > dPolarRadius_dTheta( const T& theta, const double& e )
183 {
184 //
185 // computes the deviatoric shape roundness for a given eccentricity (e) at a certain
186 // position (theta) the numerator and denominator are stored for performance reasons, as
187 // they are also needed for the derivative dRdTheta
188 T r, dRdTheta;
189
190 if ( e >= 1.0 ) {
191 r = 1;
192 dRdTheta = 0;
193 return std::make_pair( r, dRdTheta );
194 }
195
196 const double e2 = e * e;
197 const T cosTheta = cos( theta );
198 const T cos2Theta = cosTheta * cosTheta;
199
200 T numerator = ( 4 * ( 1 - e2 ) * cos2Theta + ( 2 * e - 1 ) * ( 2 * e - 1 ) );
201 T denominator = ( 2 * ( 1 - e2 ) * cosTheta +
202 ( 2 * e - 1 ) * sqrt( 4 * ( 1 - e2 ) * cos2Theta + 5 * e2 - 4 * e ) );
203
204 r = numerator / denominator;
205
206 // compute the derivative
207 const T sinTheta = sin( theta );
208
209 const T a = numerator;
210 const T b = 1. / denominator;
211 const T aux = 4 * ( 1 - e2 ) * cos2Theta + 5 * e2 - 4 * e;
212 const T dAuxdTheta = 4 * ( 1 - e2 ) * 2 * cosTheta * -sinTheta;
213 const T dadTheta = 2 * cosTheta * ( -sinTheta ) * 4 * ( 1 - e2 );
214 const T dbdTheta = -pow( denominator, -2 ) *
215 ( 2 * ( 1 - e2 ) * -sinTheta + ( 2 * e - 1 ) * 1. / 2 * pow( aux, -1. / 2 ) * dAuxdTheta );
216 dRdTheta = b * dadTheta + a * dbdTheta;
217
218 return { r, dRdTheta };
219 }

229 template < typename T >
230 T yieldFunction( const ContinuumMechanics::HaighWestergaard::HaighWestergaardCoordinates< T >& hw,
231 const double varEps = 0.0 ) const
232 {
233 const T r_ = polarRadius( hw.theta, param.e );
234 if ( varEps == 0 )
235 return ( param.Af * hw.rho ) * ( param.Af * hw.rho ) +
236 param.m * ( param.Bf * hw.rho * r_ + param.Cf * hw.xi ) - 1;
237 else
238 return param.Af * param.Af * hw.rho * hw.rho +
239 param.m * ( std::sqrt( param.Bf * hw.rho * r_ * param.Bf * hw.rho * r_ + varEps * varEps ) +
240 param.Cf * hw.xi ) -
241 1;
242 }

251 template < typename T >
252 std::tuple< T, T, T > dYieldFunction_dHaighWestergaard(
253 const ContinuumMechanics::HaighWestergaard::HaighWestergaardCoordinates< T >& hw,
254 const double varEps = 0.0 ) const
255 {
256 const auto [r_, dRdTheta_] = dPolarRadius_dTheta( hw.theta, param.e );
257
258 T dFdXi, dFdRho, dFdTheta;
259 dFdXi = param.m * param.Cf;
260
261 if ( varEps == 0.0 ) {
262 dFdRho = 2 * param.Af * hw.rho * param.Af + param.m * param.Bf * r_;
263 dFdTheta = param.m * param.Bf * hw.rho * dRdTheta_;
264 }
265 else {
266 const T auxTerm1 = param.m * 0.5 *
267 pow( param.Bf * param.Bf * hw.rho * hw.rho * r_ * r_ + varEps * varEps, -0.5 ) * 2 *
268 param.Bf * hw.rho * r_ * param.Bf;
269
270 dFdRho = param.Af * param.Af * 2 * hw.rho + auxTerm1 * r_;
271 dFdTheta = auxTerm1 * hw.rho * dRdTheta_;
272 }
273
274 return { dFdXi, dFdRho, dFdTheta };
275 }
