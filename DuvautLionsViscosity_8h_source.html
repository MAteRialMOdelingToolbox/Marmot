58 template < int s >
59 double DuvautLionsViscosity< s >::applyViscosityOnStateVar( double stateVarTrial,
60 double StateVarInf,
61 double dT )
62 {
63 return ( stateVarTrial + ( dT / viscosity ) * StateVarInf ) / ( dT / viscosity + 1 );
64 }
65
66 template < int s >
67 Marmot::Vector6d DuvautLionsViscosity< s >::applyViscosityOnStress( const Marmot::Vector6d& trialStress,
68 const Marmot::Vector6d& stressInf,
69 double dT )
70 {
71 return ( trialStress + ( dT / viscosity ) * stressInf ) / ( dT / viscosity + 1 );
72 }
73
74 template < int s >
75 typename DuvautLionsViscosity< s >::TangentSizedMatrix DuvautLionsViscosity< s >::applyViscosityOnMatTangent(
76 const TangentSizedMatrix& matTangentInv,
77 double dT )
78 {
79 return ( 1 / ( 1 + dT / viscosity ) ) * ( TangentSizedMatrix::Identity() + dT / viscosity * matTangentInv );
