102  {
103  return currentProgress >= 1.0;
104  }

106  template < int s >
107  double PerezFougetSubstepperTime< s >::getNextSubstep()
108  {
109  if ( passedSubsteps >= nPassesToIncrease )
110  currentSubstepSize *= scaleUpFactor;
111 
112  const double remainingProgress = 1.0 - currentProgress;
113  if ( remainingProgress < currentSubstepSize )
114  currentSubstepSize = remainingProgress;
115 
116  passedSubsteps++;
117  currentProgress += currentSubstepSize;
118 
119  return currentSubstepSize;
120  }
121 
122  template < int s >
123  double PerezFougetSubstepperTime< s >::getFinishedProgress()
124  {
125  return currentProgress - currentSubstepSize;
126  }
127 
128  template < int s >
129  bool PerezFougetSubstepperTime< s >::decreaseSubstepSize()
130  {
131  currentProgress -= currentSubstepSize;
132  passedSubsteps = 0;
133 
134  currentSubstepSize *= scaleDownFactor;
135 
136  if ( currentSubstepSize < minimumStepSize )
137  return MarmotJournal::warningToMSG( "UMAT: Substepper: Minimal stepzsize reached" );
138  else
139  return MarmotJournal::notificationToMSG( "UMAT: Substepper: Decreasing stepsize" );
140  }
