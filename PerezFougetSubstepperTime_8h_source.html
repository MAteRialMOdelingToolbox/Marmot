81 PerezFougetSubstepperTime< s >::PerezFougetSubstepperTime( double initialStepSize,
82 double minimumStepSize,
83 double scaleUpFactor,
84 double scaleDownFactor,
85 int nPassesToIncrease )
86 : initialStepSize( initialStepSize ),
87 minimumStepSize( minimumStepSize ),
88 scaleUpFactor( scaleUpFactor ),
89 scaleDownFactor( scaleDownFactor ),
90 nPassesToIncrease( nPassesToIncrease ),
91 currentProgress( 0.0 ),
92 currentSubstepSize( initialStepSize ),
93 passedSubsteps( 0 )
94
95 {
96 elasticTangent = TangentSizedMatrix::Identity();
97 consistentTangent = TangentSizedMatrix::Zero();
98 }
99
100 template < int s >
101 bool PerezFougetSubstepperTime< s >::isFinished()
102 {
103 return currentProgress >= 1.0;
104 }
105
106 template < int s >
107 double PerezFougetSubstepperTime< s >::getNextSubstep()
108 {
109 if ( passedSubsteps >= nPassesToIncrease )
110 currentSubstepSize *= scaleUpFactor;
111
112 const double remainingProgress = 1.0 - currentProgress;
113 if ( remainingProgress < currentSubstepSize )
114 currentSubstepSize = remainingProgress;
115
116 passedSubsteps++;
117 currentProgress += currentSubstepSize;
118
119 return currentSubstepSize;
120 }
121
122 template < int s >
123 double PerezFougetSubstepperTime< s >::getFinishedProgress()
124 {
125 return currentProgress - currentSubstepSize;
126 }
127
128 template < int s >
129 bool PerezFougetSubstepperTime< s >::decreaseSubstepSize()
130 {
131 currentProgress -= currentSubstepSize;
132 passedSubsteps = 0;
133
134 currentSubstepSize *= scaleDownFactor;
135
136 if ( currentSubstepSize < minimumStepSize )
137 return MarmotJournal::warningToMSG( "UMAT: Substepper: Minimal stepzsize reached" );
138 else
139 return MarmotJournal::notificationToMSG( "UMAT: Substepper: Decreasing stepsize" );
140 }
141
142 template < int s >
143 void PerezFougetSubstepperTime< s >::extendConsistentTangent( const Matrix6d& CelT )
144 {
145
146 elasticTangent.topLeftCorner( 6, 6 ) = CelT;
147 consistentTangent += currentSubstepSize * elasticTangent;
148 }
149
150 template < int s >
151 void PerezFougetSubstepperTime< s >::extendConsistentTangent( const Matrix6d& CelT,
152 const TangentSizedMatrix& matTangent )
153 {
154 extendConsistentTangent( CelT );
155 consistentTangent.applyOnTheLeft( matTangent );
156 }
157
158 template < int s >
159 Matrix6d PerezFougetSubstepperTime< s >::consistentStiffness()
160 {
161 return consistentTangent.topLeftCorner( 6, 6 );
162 }
