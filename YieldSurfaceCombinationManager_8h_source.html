57  template < int n >
58  YieldSurfaceCombinationManager< n >::YieldSurfaceCombinationManager() : idxUsedFlag( n )
59  {
60  yieldSurfaceCombinations.setZero();
61  const int numRows = ( 1 << n ) - 1;
62  const int numCols = n;
63 
64  for ( int row = 0; row < numRows; row++ )
65  for ( int col = 0; col < numCols; col++ )
66  yieldSurfaceCombinations( row, col ) = ( row + 1 ) & 1 << ( col );
67  return;
68  }

70  template < int n >
71  bool YieldSurfaceCombinationManager< n >::getAnotherYieldFlagCombination( YieldSurfFlagArr& activeSurfaces )
72  {
73  for ( int i = 0; i < yieldSurfaceCombinations.rows(); i++ ) {
74  bool alreadyUsed = yieldSurfaceCombinations.row( i )( idxUsedFlag );
75  YieldSurfFlagArr combinationCandidate = yieldSurfaceCombinations.row( i ).head( n );
76  if ( !alreadyUsed && ( combinationCandidate == true ).any() ) {
77  activeSurfaces = combinationCandidate;
78  return true;
79  }
80  }
81  return false;
82  }

90  template < int n >
91  void YieldSurfaceCombinationManager< n >::markYieldFlagCombinationAsUsed(
92  const YieldSurfFlagArr& activeSurfaces )
93  {
94  for ( int i = 0; i < yieldSurfaceCombinations.rows(); i++ )
95  if ( ( yieldSurfaceCombinations.row( i ).head( n ) == activeSurfaces ).all() )
