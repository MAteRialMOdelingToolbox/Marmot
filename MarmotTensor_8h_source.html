53  template <int sizeI, int sizeJ>
54  Eigen::Matrix<double, sizeI * sizeJ, sizeI * sizeJ> makeIndexSwapTensor()
55  {
56  // Aux. Matrix, which helps to swap indices in Eigen::Matrices abused as higher order Tensors by
57  // multiplication ,
58  //
59  // T_(ij)(kl) * IndexSwapTensor_(kl)(lk) = T_(ij)(lk)
60  //
61  // For instance:
62  //
63  // Matrix3d t;
64  // t<<1,2,3,4,5,6,7,8,9;
65  // const auto P = makeIndexSwapTensor<3,3>();
66  // std::cout << t.reshaped().transpose() << std::endl;
67  // std::cout << t.reshaped().transpose() * P << std::endl;
68  //
69  // Output:
70  //
71  // 1,4,7,2,5,8,3,6,9
72  // 1,2,3,4,5,6,7,8,9
73  //
74 
75  Eigen::Matrix<double, sizeI * sizeJ, sizeI * sizeJ> P;
76 
77  auto d = []( int a, int b ) -> double { return a == b ? 1.0 : 0.0; };
78 
79  for ( int i = 0; i < sizeI; i++ )
80  for ( int j = 0; j < sizeJ; j++ )
81  for ( int l = 0; l < sizeI; l++ )
82  for ( int k = 0; k < sizeJ; k++ )
83  P( i + j * sizeI, l * sizeJ + k ) = d( i, l ) * d( k, j );
84 
85  return P;
86  }
