56  template < int sizeI, int sizeJ >
57  Eigen::Matrix< double, sizeI * sizeJ, sizeI * sizeJ > makeIndexSwapTensor()
58  {
59  // Aux. Matrix, which helps to swap indices in Eigen::Matrices abused as higher order Tensors by
60  // multiplication ,
61  //
62  // T_(ij)(kl) * IndexSwapTensor_(kl)(lk) = T_(ij)(lk)
63  //
64  // For instance:
65  //
66  // Matrix3d t;
67  // t<<1,2,3,4,5,6,7,8,9;
68  // const auto P = makeIndexSwapTensor<3,3>();
69  // std::cout << t.reshaped().transpose() << std::endl;
70  // std::cout << t.reshaped().transpose() * P << std::endl;
71  //
72  // Output:
73  //
74  // 1,4,7,2,5,8,3,6,9
75  // 1,2,3,4,5,6,7,8,9
76  //
77 
78  Eigen::Matrix< double, sizeI * sizeJ, sizeI * sizeJ > P;
79 
80  auto d = []( int a, int b ) -> double { return a == b ? 1.0 : 0.0; };
81 
82  for ( int i = 0; i < sizeI; i++ )
83  for ( int j = 0; j < sizeJ; j++ )
84  for ( int l = 0; l < sizeI; l++ )
85  for ( int k = 0; k < sizeJ; k++ )
86  P( i + j * sizeI, l * sizeJ + k ) = d( i, l ) * d( k, j );
87 
88  return P;
89  }

150  {
151  if constexpr ( nDim == 1 )
152  return std::pair< int, int >( 0, 0 );
153  else if ( nDim == 2 )
154  switch ( ij ) {
155  case 0: return std::pair< int, int >( 0, 0 );
156  case 1: return std::pair< int, int >( 1, 1 );
157  case 2: return std::pair< int, int >( 0, 1 );
158  }
159 
160  else if ( nDim == 3 ) {
161  switch ( ij ) {
162  case 0: return std::pair< int, int >( 0, 0 );
163  case 1: return std::pair< int, int >( 1, 1 );
164  case 2: return std::pair< int, int >( 2, 2 );
165  case 3: return std::pair< int, int >( 0, 1 );
166  case 4: return std::pair< int, int >( 0, 2 );
167  case 5: return std::pair< int, int >( 1, 2 );
