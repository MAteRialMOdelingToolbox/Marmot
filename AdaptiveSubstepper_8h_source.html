99 template <size_t n, size_t nState>
100 AdaptiveSubstepper<n, nState>::AdaptiveSubstepper( double initialStepSize,
101 double minimumStepSize,
102 double maxScaleUpFactor,
103 double scaleDownFactor,
104 double integrationErrorTolerance,
105 int nPassesToIncrease,
106 const Matrix6d& Cel )
107 : initialStepSize( initialStepSize ),
108 minimumStepSize( minimumStepSize ),
109 maxScaleUpFactor( maxScaleUpFactor ),
110 scaleDownFactor( scaleDownFactor ),
111 integrationErrorTolerance( integrationErrorTolerance ),
112 nPassesToIncrease( nPassesToIncrease ),
113 ignoreErrorToleranceOnMinimumStepSize( true ),
114 currentProgress( 0.0 ),
115 currentSubstepSize( initialStepSize ),
116 passedSubsteps( 0 ),
117 substepIndex( -1 )
118 {
119 elasticTangent = TangentSizedMatrix::Identity();
120 elasticTangent.topLeftCorner( 6, 6 ) = Cel;
121
122 consistentTangentProgress = TangentSizedMatrix::Zero();
123 consistentTangentProgressFullTemp = TangentSizedMatrix::Zero();
124 consistentTangentProgressHalfTemp = TangentSizedMatrix::Zero();
125
126 stressProgressHalfTemp.setZero();
127 stressProgressFullTemp.setZero();
128 stateProgressHalfTemp.setZero();
129 stateProgressFullTemp.setZero();
130 consistentTangentProgressHalfTemp.setZero();
131 consistentTangentProgressFullTemp.setZero();
132
133 currentState = FullStep;
134 }

144 template <size_t n, size_t nState>
145 bool AdaptiveSubstepper<n, nState>::isFinished()
146 {
147 return ( ( 1.0 - currentProgress ) <= 2e-16 && currentState == FullStep );
148 }

150 template <size_t n, size_t nState>
151 double AdaptiveSubstepper<n, nState>::getNextSubstep()
152 {
153 switch ( currentState ) {
154 case FullStep: {
155 const double remainingProgress = 1.0 - currentProgress;
156 if ( remainingProgress < currentSubstepSize )
157 currentSubstepSize = remainingProgress;
158 substepIndex++;
159 return currentSubstepSize;
160 break;
161 }
162 case FirstHalfStep: {
163 return 0.5 * currentSubstepSize;
164 break;
165 }
166 case SecondHalfStep: {
167 return 0.5 * currentSubstepSize;
168 break;
169 }
170 default: return -1.0;
171 }
172 }

174 void AdaptiveSubstepper<n, nState>::getConvergedProgress( Marmot::Vector6d& stress, IntegrationStateVector& stateVars )
175 {
176 switch ( currentState ) {
177 case FullStep: {
178 stress = stressProgress;
179 stateVars = stateProgress;
180 break;
181 }
182 case FirstHalfStep: {
183 stress = stressProgress;
184 stateVars = stateProgress;

186 }
187 case SecondHalfStep: {
188 stress = stressProgressHalfTemp;
189 stateVars = stateProgressHalfTemp;
190 break;

195 template <size_t n, size_t nState>
196 bool AdaptiveSubstepper<n, nState>::discardSubstep()
197 {
198 passedSubsteps = 0;
199 switch ( currentState ) {
200 case FullStep: {
201 currentSubstepSize *= scaleDownFactor; // we use the scale factor only here
202 break;
203 }
204 // these cases should actually never happen, as the full step has already converged!
205 case FirstHalfStep:
206 MarmotJournal::warningToMSG( "UMAT: warning, 1th half sub step has not converged after already "
207 "converged full step" );
208 return acceptSubstepWithFullStepOnly();
209
210 case SecondHalfStep:
211 MarmotJournal::warningToMSG( "UMAT: warning, 2th half sub step has not converged after already "
212 "converged full step" );
213 return acceptSubstepWithFullStepOnly();
214 }
215
216 currentState = FullStep;
217
218 if ( currentSubstepSize < minimumStepSize )
219 return MarmotJournal::warningToMSG( "UMAT: Substepper: Minimal stepzsize reached" );
220 else
221 return true;
222 }
223
224 template <size_t n, size_t nState>
225 bool AdaptiveSubstepper<n, nState>::repeatSubstep( double factorNew )
226 {
227 currentState = FullStep;
228 passedSubsteps = 0;
229
230 currentSubstepSize *= factorNew; // we use the scale factor only here
231
232 if ( currentSubstepSize < minimumStepSize )
233 return MarmotJournal::warningToMSG( "UMAT: Substepper: Minimal stepzsize reached" );
234 else
235 return true;
236 }

238 template <size_t n, size_t nState>
239 bool AdaptiveSubstepper<n, nState>::finishSubstep( const Marmot::Vector6d& resultStress,
240 const TangentSizedMatrix& dXdY,
241 const IntegrationStateVector& stateVars )
242 {
243 if ( currentState == FullStep ) {
244 stressProgressFullTemp = resultStress;
245 stateProgressFullTemp = stateVars;
246 consistentTangentProgressFullTemp = consistentTangentProgress;
247 consistentTangentProgressFullTemp += currentSubstepSize * elasticTangent;
248 consistentTangentProgressFullTemp.applyOnTheLeft( dXdY );
249 currentState = FirstHalfStep;
250 return true;
251 }
252 else if ( currentState == FirstHalfStep ) {
253
254 stressProgressHalfTemp = resultStress;
255 stateProgressHalfTemp = stateVars;
256 consistentTangentProgressHalfTemp = consistentTangentProgress;
257 consistentTangentProgressHalfTemp += 0.5 * currentSubstepSize * elasticTangent;
258 consistentTangentProgressHalfTemp.applyOnTheLeft( dXdY );
259 currentState = SecondHalfStep;
260 return true;
261 }
262
263 else if ( currentState == SecondHalfStep ) {
264 // error Estimation
265
266 currentState = FullStep;
267 const double error = ( resultStress - stressProgressFullTemp ).norm();
268 const double errorRatio = error / integrationErrorTolerance;
269 double scaleFactor = 1.0;
270 if ( errorRatio > 1e-10 )
271 scaleFactor = 0.9 * std::sqrt( 1. / errorRatio );
272
273 // saturations
274 if ( scaleFactor < 1e-1 )
275 scaleFactor = 1e-1;
276 if ( scaleFactor * currentSubstepSize < minimumStepSize )
277 scaleFactor = minimumStepSize / currentSubstepSize;
278 if ( scaleFactor > maxScaleUpFactor )
279 scaleFactor = maxScaleUpFactor;
280 if ( scaleFactor > 10 )
281 scaleFactor = 10;
282
283 // Error large than tolerance?
284 if ( error > integrationErrorTolerance ) {
285 passedSubsteps = 0;
286 if ( errorRatio < 2 ) {
287 return splitCurrentSubstep();
288 }
289 else {
290 return repeatSubstep( scaleFactor );
291 }
292 }
293 else {
294 stressProgressHalfTemp = resultStress;
295 stateProgressHalfTemp = stateVars;
296 consistentTangentProgressHalfTemp += 0.5 * currentSubstepSize * elasticTangent;
297 consistentTangentProgressHalfTemp.applyOnTheLeft( dXdY );
298
299 consistentTangentProgress = 2 * consistentTangentProgressHalfTemp - consistentTangentProgressFullTemp;
300 stressProgress = 2 * stressProgressHalfTemp - stressProgressFullTemp;
301 stateProgress = 2 * stateProgressHalfTemp - stateProgressFullTemp;
302
303 currentProgress += currentSubstepSize;
304
305 passedSubsteps++;
306 currentSubstepSize *= scaleFactor;
307
308 return true;
309 }
310 }
311 else
312 return false;
313 }

315 template <size_t n, size_t nState>
316 void AdaptiveSubstepper<n, nState>::finishElasticSubstep( const Marmot::Vector6d& newStress )
317 {
318 switch ( currentState ) {
319 case FullStep: {
320 // this means, that the complete current cycle is already successfull,
321 // as the two half steps must also be elastic!
322 consistentTangentProgress += currentSubstepSize * elasticTangent;
323 stressProgress = newStress;
324 // no need for two half steps if full step was already elastic
325 currentProgress += currentSubstepSize;
326 currentState = FullStep;
327 passedSubsteps++;
328 break;
329 }
330
331 case FirstHalfStep: {
332 consistentTangentProgressHalfTemp += 0.5 * currentSubstepSize * elasticTangent;
333 stressProgressHalfTemp = newStress;
334 currentState = SecondHalfStep;
335 break;
336 }
337 case SecondHalfStep: {
338 acceptSubstepWithFullStepOnly();
339 break;
340 }
341 }
342 }

345 void AdaptiveSubstepper<n, nState>::getResults( Marmot::Vector6d& stress,

347 IntegrationStateVector& stateVars )

350 stateVars = stateProgress;
351 consistentTangentOperator = consistentTangentProgress.topLeftCorner( 6, 6 );

354 template <size_t n, size_t nState>
355 bool AdaptiveSubstepper<n, nState>::acceptSubstepWithFullStepOnly()
356 {
357 consistentTangentProgress = consistentTangentProgressFullTemp;
358 stressProgress = stressProgressFullTemp;
359 stateProgress = stateProgressFullTemp;
360
361 currentProgress += currentSubstepSize;
362 currentState = FullStep;
363
364 return true;
365 }

367 template <size_t n, size_t nState>
368 bool AdaptiveSubstepper<n, nState>::splitCurrentSubstep()
369 {
370 if ( currentSubstepSize < 2 * minimumStepSize ) {
371 if ( ignoreErrorToleranceOnMinimumStepSize )
372 return acceptSubstepWithFullStepOnly();
373 else
374 return false;
375 }
376
377 consistentTangentProgressFullTemp = consistentTangentProgressHalfTemp;
378 stressProgressFullTemp = stressProgressHalfTemp;
379 stateProgressFullTemp = stateProgressHalfTemp;
380 currentSubstepSize *= 0.5;
381 currentState = FirstHalfStep;
382
383 return true;
384 }
385 template <size_t n, size_t nState>
386 int AdaptiveSubstepper<n, nState>::getNumberOfSubsteps()
387 {
388 return substepIndex;
