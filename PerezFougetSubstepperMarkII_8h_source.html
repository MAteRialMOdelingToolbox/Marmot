95 template < int n >
96 bool PerezFougetSubstepper< n >::isFinished()
97 {
98 // this is due to numerical accuracy ...
99 return ( 1.0 - currentProgress ) <= 2e-16;
100 }
101
102 template < int n >
103 double PerezFougetSubstepper< n >::getNextSubstep()
104 {
105 if ( passedSubsteps >= nPassesToIncrease )
106 currentSubstepSize *= scaleUpFactor;
107
108 const double remainingProgress = 1.0 - currentProgress;
109 if ( remainingProgress < currentSubstepSize )
110 currentSubstepSize = remainingProgress;
111
112 passedSubsteps++;
113 currentProgress += currentSubstepSize;
114
115 return currentSubstepSize;
116 }
117
118 template < int n >
119 bool PerezFougetSubstepper< n >::decreaseSubstepSize()
120 {
121 currentProgress -= currentSubstepSize;
122 passedSubsteps = 0;
123
124 currentSubstepSize *= scaleDownFactor;
125
126 if ( currentSubstepSize < minimumStepSize )
127 return MarmotJournal::warningToMSG( "UMAT: Substepper: Minimal stepzsize reached" );
128 else
129 return MarmotJournal::notificationToMSG( "UMAT: Substepper: Decreasing stepsize" );
130 }
131
132 template < int n >
133 void PerezFougetSubstepper< n >::finishElasticSubstep()
134 {
135 consistentTangent += currentSubstepSize * TangentSizedMatrix::Identity();
136 }
137
138 template < int n >
139 void PerezFougetSubstepper< n >::finishSubstep( const TangentSizedMatrix& dXdY )
140 {
141 finishElasticSubstep();
142 consistentTangent.applyOnTheLeft( dXdY );
143
144 for ( int i = 0; i < consistentTangent.rows(); i++ )
145 for ( int j = 0; j < consistentTangent.cols(); j++ )
146 if ( std::abs( consistentTangent( i, j ) ) < 1e-12 )
147 consistentTangent( i, j ) = 0.0;
148 }
149
150 template < int n >
151 Matrix6d PerezFougetSubstepper< n >::consistentStiffness()
152 {
153 return consistentTangent.topLeftCorner( 6, 6 ) * Cel;
154 }
