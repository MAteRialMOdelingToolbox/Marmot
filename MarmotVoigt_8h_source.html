51 template < int voigtSize >
52 Eigen::Matrix< double, voigtSize, 1 > reduce3DVoigt( const Marmot::Vector6d& Voigt3D )
53 {
54 if constexpr ( voigtSize == 1 )
55 return ( Eigen::Matrix< double, 1, 1 >() << Voigt3D( 0 ) ).finished();
56 else if constexpr ( voigtSize == 3 )
57 return voigtToPlaneVoigt( Voigt3D );
58 else if constexpr ( voigtSize == 6 )
59 return Voigt3D;
60 else
61 throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
62 }
63
64 template < int voigtSize >
65 Marmot::Vector6d make3DVoigt( const Eigen::Matrix< double, voigtSize, 1 >& Voigt )
66 {
67 if constexpr ( voigtSize == 1 )
68 return ( Marmot::Vector6d() << Voigt( 0 ), 0, 0, 0, 0, 0 ).finished();
69 else if constexpr ( voigtSize == 3 )
70 return planeVoigtToVoigt( Voigt );
71 else if constexpr ( voigtSize == 6 )
72 return Voigt;
73 else
74 throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
75 }

83 template < int nDim >
84 Eigen::Matrix< double, nDim, nDim > stressMatrixFromVoigt(
85 const Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >& Voigt )
86 {
87 if constexpr ( nDim == 1 )
88 return ( Eigen::Matrix< double, nDim, nDim >() << Voigt( 0 ) ).finished();
89 else if constexpr ( nDim == 2 )
90 return ( Eigen::Matrix< double, nDim, nDim >() << Voigt( 0 ), Voigt( 2 ), Voigt( 2 ), Voigt( 1 ) )
91 .finished();
92 else if constexpr ( nDim == 3 )
93 return voigtToStress( Voigt );
94 else
95 throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
96 }
97
98 template < int nDim >
99 Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 > voigtFromStrainMatrix(
100 const Eigen::Matrix< double, nDim, nDim >& strain )
101 {
102 if constexpr ( nDim == 1 )
103 return ( Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >() << strain( 0, 0 ) ).finished();
104 else if constexpr ( nDim == 2 )
105 return ( Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >() << strain( 0, 0 ),
106 strain( 1, 1 ),
107 2 * strain( 0, 1 ) )
108 .finished();
109 else if constexpr ( nDim == 3 )
110 return strainToVoigt( strain );
111 else
112 throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
113 }
