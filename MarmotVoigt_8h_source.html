53  template <int voigtSize>
54  Eigen::Matrix<double, voigtSize, 1> reduce3DVoigt( const Marmot::Vector6d& Voigt3D )
55  {
56  if constexpr ( voigtSize == 1 )
57  return ( Eigen::Matrix<double, 1, 1>() << Voigt3D( 0 ) ).finished();
58  else if constexpr ( voigtSize == 3 )
59  return voigtToPlaneVoigt( Voigt3D );
60  else if constexpr ( voigtSize == 6 )
61  return Voigt3D;
62  else
63  throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
64  }

97  template <int nDim>
98  Eigen::Matrix<double, nDim, nDim> StressMatrixFromVoigt(
99  const Eigen::Matrix<double, VOIGTFROMDIM( nDim ), 1>& Voigt )
100  {
101  if constexpr ( nDim == 1 )
102  return ( Eigen::Matrix<double, nDim, nDim>() << Voigt( 0 ) ).finished();
103  else if constexpr ( nDim == 2 )
104  return ( Eigen::Matrix<double, nDim, nDim>() << Voigt( 0 ), Voigt( 2 ), Voigt( 2 ), Voigt( 1 ) )
105  .finished();
106  else if constexpr ( nDim == 3 )
107  return voigtToStress( Voigt );
108  else
109  throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
110  }

112  template <int nDim>
113  Eigen::Matrix<double, VOIGTFROMDIM( nDim ), 1> VoigtFromStrainMatrix(
114  const Eigen::Matrix<double, nDim, nDim>& strain )
115  {
116  if constexpr ( nDim == 1 )
117  return ( Eigen::Matrix<double, VOIGTFROMDIM( nDim ), 1>() << strain( 0, 0 ) ).finished();
118  else if constexpr ( nDim == 2 )
119  return ( Eigen::Matrix<double, VOIGTFROMDIM( nDim ), 1>() << strain( 0, 0 ),
120  strain( 1, 1 ),
121  2 * strain( 0, 1 ) )
122  .finished();
123  else if constexpr ( nDim == 3 )
124  return strainToVoigt( strain );
125  else
126  throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
127  }
