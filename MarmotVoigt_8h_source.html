237 template < typename T = double >
238 Eigen::Matrix< T, 6, 1 > stressToVoigt( const Eigen::Matrix< T, 3, 3 >& stressTensor )
239 {
240 Eigen::Matrix< T, 6, 1 > stress;
241 // clang-format off
242 stress << stressTensor( 0, 0 ),
243 stressTensor( 1, 1 ),
244 stressTensor( 2, 2 ),
245 stressTensor( 0, 1 ),
246 stressTensor( 0, 2 ),
247 stressTensor( 1, 2 );
248 // clang-format on
249 return stress;
250 }

252 template < int nDim >
253 Eigen::Matrix< double, nDim, nDim > stressMatrixFromVoigt(
254 const Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >& Voigt )
255 {
256 if constexpr ( nDim == 1 )
257 return ( Eigen::Matrix< double, nDim, nDim >() << Voigt( 0 ) ).finished();
258 else if constexpr ( nDim == 2 )
259 return ( Eigen::Matrix< double, nDim, nDim >() << Voigt( 0 ), Voigt( 2 ), Voigt( 2 ), Voigt( 1 ) ).finished();
260 else if constexpr ( nDim == 3 )
261 return voigtToStress( Voigt );
262 else
263 throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
264 }

266 template < int nDim >
267 Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 > voigtFromStrainMatrix(
268 const Eigen::Matrix< double, nDim, nDim >& strain )
269 {
270 if constexpr ( nDim == 1 )
271 return ( Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >() << strain( 0, 0 ) ).finished();
272 else if constexpr ( nDim == 2 )
273 return ( Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >() << strain( 0, 0 ),
274 strain( 1, 1 ),
275 2 * strain( 0, 1 ) )
276 .finished();
277 else if constexpr ( nDim == 3 )
278 return strainToVoigt( strain );
279 else
280 throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
281 }

391 template < typename T >
392 T I2( const Eigen::Matrix< T, 6, 1 >& stress )
393 {
394 const Eigen::Matrix< T, 6, 1 >& s = stress;
395 return s( 0 ) * s( 1 ) + s( 1 ) * s( 2 ) + s( 2 ) * s( 0 ) - s( 3 ) * s( 3 ) - s( 4 ) * s( 4 ) -
396 s( 5 ) * s( 5 );
397 }

412 template < typename T >
413 T I3( const Eigen::Matrix< T, 6, 1 >& stress )
414 {
415 const Eigen::Matrix< T, 6, 1 >& s = stress;
416 return s( 0 ) * s( 1 ) * s( 2 ) + 2. * s( 3 ) * s( 4 ) * s( 5 ) - s( 0 ) * s( 5 ) * s( 5 ) -
417 s( 1 ) * s( 4 ) * s( 4 ) - s( 2 ) * s( 3 ) * s( 3 );
418 }

430 template < typename T >
431 T J2( const Eigen::Matrix< T, 6, 1 >& stress )
432 {
433 auto I1_ = I1( stress );
434 auto I2_ = I2( stress );
435 auto res = ( 1. / 3 ) * I1_ * I1_ - I2_;
436 return Marmot::Math::makeReal( res ) >= 0 ? res : T( 0.0 );
437 }

453 template < typename T >
454 T J3( const Eigen::Matrix< T, 6, 1 >& stress )
455 {
456 T I1_ = I1( stress );
457 T I2_ = I2( stress );
458 T I3_ = I3( stress );
459
460 return ( 2. / 27 ) * pow( I1_, 3 ) - ( 1. / 3 ) * I1_ * I2_ + I3_;
461 }

487 template < typename T >
488 Eigen::Matrix< T, 6, 1 > dRho_dStress( T rho, const Eigen::Matrix< T, 6, 1 >& stress )
489 {
490 if ( Marmot::Math::makeReal( rho ) <= 1e-16 )
491 return Eigen::Matrix< T, 6, 1 >::Zero();
492
493 Eigen::Matrix< T, 6, 1 > s = IDev * stress;
494
495 return T( 1. / rho ) * P.array() * s.array();
496 }
