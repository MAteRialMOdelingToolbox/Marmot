108 template < int voigtSize >
109 Eigen::Matrix< double, voigtSize, 1 > reduce3DVoigt( const Marmot::Vector6d& Voigt3D )
110 {
111 if constexpr ( voigtSize == 1 )
112 return ( Eigen::Matrix< double, 1, 1 >() << Voigt3D( 0 ) ).finished();
113 else if constexpr ( voigtSize == 3 )
114 return voigtToPlaneVoigt( Voigt3D );
115 else if constexpr ( voigtSize == 6 )
116 return Voigt3D;
117 else
118 throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
119 }

142 {
143 if constexpr ( voigtSize == 1 )

231 template < int nDim >
232 Eigen::Matrix< double, nDim, nDim > stressMatrixFromVoigt(
233 const Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >& Voigt )
234 {
235 if constexpr ( nDim == 1 )
236 return ( Eigen::Matrix< double, nDim, nDim >() << Voigt( 0 ) ).finished();
237 else if constexpr ( nDim == 2 )
238 return ( Eigen::Matrix< double, nDim, nDim >() << Voigt( 0 ), Voigt( 2 ), Voigt( 2 ), Voigt( 1 ) ).finished();
239 else if constexpr ( nDim == 3 )
240 return voigtToStress( Voigt );
241 else
242 throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
243 }
244
245 template < int nDim >
246 Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 > voigtFromStrainMatrix(
247 const Eigen::Matrix< double, nDim, nDim >& strain )
248 {
249 if constexpr ( nDim == 1 )
250 return ( Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >() << strain( 0, 0 ) ).finished();
251 else if constexpr ( nDim == 2 )
252 return ( Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >() << strain( 0, 0 ),
253 strain( 1, 1 ),
254 2 * strain( 0, 1 ) )
255 .finished();
256 else if constexpr ( nDim == 3 )
257 return strainToVoigt( strain );
258 else
259 throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
260 }
