53  template < int voigtSize >
54  Eigen::Matrix< double, voigtSize, 1 > reduce3DVoigt( const Marmot::Vector6d& Voigt3D )
55  {
56  if constexpr ( voigtSize == 1 )
57  return ( Eigen::Matrix< double, 1, 1 >() << Voigt3D( 0 ) ).finished();
58  else if constexpr ( voigtSize == 3 )
59  return voigtToPlaneVoigt( Voigt3D );
60  else if constexpr ( voigtSize == 6 )
61  return Voigt3D;
62  else
63  throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
64  }

85  template < int nDim >
86  Eigen::Matrix< double, nDim, nDim > StressMatrixFromVoigt(
87  const Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >& Voigt )
88  {
89  if constexpr ( nDim == 1 )
90  return ( Eigen::Matrix< double, nDim, nDim >() << Voigt( 0 ) ).finished();
91  else if constexpr ( nDim == 2 )
92  return ( Eigen::Matrix< double, nDim, nDim >() << Voigt( 0 ), Voigt( 2 ), Voigt( 2 ), Voigt( 1 ) )
93  .finished();
94  else if constexpr ( nDim == 3 )
95  return voigtToStress( Voigt );
96  else
97  throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
98  }

100  template < int nDim >
101  Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 > VoigtFromStrainMatrix(
102  const Eigen::Matrix< double, nDim, nDim >& strain )
103  {
104  if constexpr ( nDim == 1 )
105  return ( Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >() << strain( 0, 0 ) ).finished();
106  else if constexpr ( nDim == 2 )
107  return ( Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >() << strain( 0, 0 ),
108  strain( 1, 1 ),
109  2 * strain( 0, 1 ) )
110  .finished();
111  else if constexpr ( nDim == 3 )
112  return strainToVoigt( strain );
113  else
114  throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
115  }
