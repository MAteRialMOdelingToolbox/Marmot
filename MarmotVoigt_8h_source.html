51  template < int voigtSize >
52  Eigen::Matrix< double, voigtSize, 1 > reduce3DVoigt( const Marmot::Vector6d& Voigt3D )
53  {
54  if constexpr ( voigtSize == 1 )
55  return ( Eigen::Matrix< double, 1, 1 >() << Voigt3D( 0 ) ).finished();
56  else if constexpr ( voigtSize == 3 )
57  return voigtToPlaneVoigt( Voigt3D );
58  else if constexpr ( voigtSize == 6 )
59  return Voigt3D;
60  else
61  throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
62  }

83  template < int nDim >
84  Eigen::Matrix< double, nDim, nDim > stressMatrixFromVoigt(
85  const Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >& Voigt )
86  {
87  if constexpr ( nDim == 1 )
88  return ( Eigen::Matrix< double, nDim, nDim >() << Voigt( 0 ) ).finished();
89  else if constexpr ( nDim == 2 )
90  return ( Eigen::Matrix< double, nDim, nDim >() << Voigt( 0 ), Voigt( 2 ), Voigt( 2 ), Voigt( 1 ) ).finished();
91  else if constexpr ( nDim == 3 )
92  return voigtToStress( Voigt );
93  else
94  throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
95  }

97  template < int nDim >
98  Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 > voigtFromStrainMatrix(
99  const Eigen::Matrix< double, nDim, nDim >& strain )
100  {
101  if constexpr ( nDim == 1 )
102  return ( Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >() << strain( 0, 0 ) ).finished();
103  else if constexpr ( nDim == 2 )
104  return ( Eigen::Matrix< double, VOIGTFROMDIM( nDim ), 1 >() << strain( 0, 0 ),
105  strain( 1, 1 ),
106  2 * strain( 0, 1 ) )
107  .finished();
108  else if constexpr ( nDim == 3 )
109  return strainToVoigt( strain );
110  else
111  throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
112  }
