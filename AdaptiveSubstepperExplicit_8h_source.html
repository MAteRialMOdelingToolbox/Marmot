181  template <size_t n, size_t nState>
182  double AdaptiveSubstepperExplicit<n, nState>::getNextSubstep()
183  {
184  switch ( currentState ) {
185  case FullStep: {
186  const double remainingProgress = 1.0 - currentProgress;
187  if ( remainingProgress < currentSubstepSize )
188  currentSubstepSize = remainingProgress;
189  substepIndex++;
190  return currentSubstepSize;
191  break;
192  }
193  case FirstHalfStep: {
194  return 0.5 * currentSubstepSize;
195  break;
196  }
197  case SecondHalfStep: {
198  return 0.5 * currentSubstepSize;
199  break;
200  }
201  }
202  return 0;
203  }

227  template <size_t n, size_t nState>
228  bool AdaptiveSubstepperExplicit<n, nState>::discardSubstep()
229  {
230  passedSubsteps = 0;
231  switch ( currentState ) {
232  case FullStep: {
233  currentSubstepSize *= scaleDownFactor; // we use the scale factor only here
234  break;
235  }
236  // these cases should actually never happen, as the full step has already converged!
237  case FirstHalfStep:
238  MarmotJournal::warningToMSG( "UMAT: warning, 1th half sub step has not converged after already "
239  "converged full step" );
240  return acceptSubstepWithFullStepOnly();
241 
242  case SecondHalfStep:
243  MarmotJournal::warningToMSG( "UMAT: warning, 2th half sub step has not converged after already "
244  "converged full step" );
245  return acceptSubstepWithFullStepOnly();
246  }
247 
248  currentState = FullStep;
249 
250  if ( currentSubstepSize < minimumStepSize )
251  return MarmotJournal::warningToMSG( "UMAT: Substepper: Minimal stepzsize reached" );
252  else
253  return true;
254  }

257  bool AdaptiveSubstepperExplicit<n, nState>::repeatSubstep( double factorNew )
258  {
259  currentState = FullStep;
260  passedSubsteps = 0;
261 
262  currentSubstepSize *= factorNew; // we use the scale factor only here
263 
264  if ( currentSubstepSize < minimumStepSize )
265  return MarmotJournal::warningToMSG( "UMAT: Substepper: Minimal stepzsize reached" );
266  else
267  return true;

270  template <size_t n, size_t nState>
271  bool AdaptiveSubstepperExplicit<n, nState>::finishSubstep( const Marmot::Vector6d& resultStress,
272  const TangentSizedMatrix& dXdY,
273  const TangentSizedMatrix& dYdXOld,
274  const IntegrationStateVector& stateVars )
275  {
276  if ( currentState == FullStep ) {
277  stressProgressFullTemp = resultStress;
278  stateProgressFullTemp = stateVars;
279  consistentTangentProgressFullTemp = consistentTangentProgress;
280  consistentTangentProgressFullTemp.applyOnTheLeft( IXX() - dYdXOld );
281  consistentTangentProgressFullTemp += currentSubstepSize * IX6() * Cel;
282  consistentTangentProgressFullTemp.applyOnTheLeft( dXdY );
283  currentState = FirstHalfStep;
284  return true;
285  }
286  else if ( currentState == FirstHalfStep ) {
287 
288  stressProgressHalfTemp = resultStress;
289  stateProgressHalfTemp = stateVars;
290  consistentTangentProgressHalfTemp = consistentTangentProgress;
291  consistentTangentProgressHalfTemp.applyOnTheLeft( IXX() - dYdXOld );
292  consistentTangentProgressHalfTemp += 0.5 * currentSubstepSize * IX6() * Cel;
293  consistentTangentProgressHalfTemp.applyOnTheLeft( dXdY );
294  currentState = SecondHalfStep;
295  return true;
296  }
297 
298  else if ( currentState == SecondHalfStep ) {
299  // error Estimation
300 
301  currentState = FullStep;
302  const double error = ( resultStress - stressProgressFullTemp ).norm();
303  const double errorRatio = error / integrationErrorTolerance;
304  double scaleFactor = 1.0;
305  if ( errorRatio > 1e-10 )
306  scaleFactor = 0.9 * std::sqrt( 1. / errorRatio );
307 
308  // saturations
309  if ( scaleFactor < 0.1 )
310  scaleFactor = 0.1;
311  if ( scaleFactor * currentSubstepSize < minimumStepSize )
312  scaleFactor = minimumStepSize / currentSubstepSize;
313  if ( scaleFactor > maxScaleUpFactor )
314  scaleFactor = maxScaleUpFactor;
315  if ( scaleFactor > 10 )
316  scaleFactor = 10;
317 
318  // Error large than tolerance?
319  if ( error > integrationErrorTolerance ) {
320  discardedDueToError++;
321  passedSubsteps = 0;
322  if ( errorRatio < 2 ) {
323  return splitCurrentSubstep();
324  }
325  else {
326  return repeatSubstep( scaleFactor );
327  }
328  // return splitCurrentSubstep(); }
329  }
330  else {
331 
332  stressProgressHalfTemp = resultStress;
333  stateProgressHalfTemp = stateVars;
334  consistentTangentProgressHalfTemp.applyOnTheLeft( IXX() - dYdXOld );
335  consistentTangentProgressHalfTemp += 0.5 * currentSubstepSize * IX6() * Cel;
336  consistentTangentProgressHalfTemp.applyOnTheLeft( dXdY );
337 
338  consistentTangentProgress = 2 * consistentTangentProgressHalfTemp - consistentTangentProgressFullTemp;
339  stressProgress = 2 * stressProgressHalfTemp - stressProgressFullTemp;
340  stateProgress = 2 * stateProgressHalfTemp - stateProgressFullTemp;
341  currentProgress += currentSubstepSize;
342 
343  passedSubsteps++;
344  currentSubstepSize *= scaleFactor;
345 
346  return true;
347  }
348  }
349  return false;
350  }

352  template <size_t n, size_t nState>
353  void AdaptiveSubstepperExplicit<n, nState>::finishElasticSubstep( const Marmot::Vector6d& newStress )
354  {
355  switch ( currentState ) {
356  case FullStep: {
357  // this means, that the complete current cycle is already successfull,
358  // as the two half steps must also be elastic!
359  consistentTangentProgress += currentSubstepSize * IX6() * Cel;
360  stressProgress = newStress;
361  // no need for two half steps if full step was already elastic
362  currentProgress += currentSubstepSize;
363  currentState = FullStep;
364  passedSubsteps++;
365  break;
366  }
367 
368  case FirstHalfStep: {
369  consistentTangentProgressHalfTemp = consistentTangentProgress;
370  consistentTangentProgressHalfTemp += 0.5 * currentSubstepSize * IX6() * Cel;
371  stressProgressHalfTemp = newStress;
372  currentState = SecondHalfStep;
373  break;
374  }
375  case SecondHalfStep: {
376  acceptSubstepWithFullStepOnly();
377  break;
378  }
379  }
380  }

405  template <size_t n, size_t nState>
406  bool AdaptiveSubstepperExplicit<n, nState>::splitCurrentSubstep()
407  {
408  if ( currentSubstepSize < 2 * minimumStepSize ) {
409  if ( ignoreErrorToleranceOnMinimumStepSize )
410  return acceptSubstepWithFullStepOnly();
411  else
412  return false;
413  }
414 
415  consistentTangentProgressFullTemp = consistentTangentProgressHalfTemp;
416  stressProgressFullTemp = stressProgressHalfTemp;
417  stateProgressFullTemp = stateProgressHalfTemp;
418  currentSubstepSize *= 0.5;
419  currentState = FirstHalfStep;
420 
421  return true;
422  }
