182 template < size_t n, size_t nState >
183 double AdaptiveSubstepperExplicit< n, nState >::getNextSubstep()
184 {
185 switch ( currentState ) {
186 case FullStep: {
187 const double remainingProgress = 1.0 - currentProgress;
188 if ( remainingProgress < currentSubstepSize )
189 currentSubstepSize = remainingProgress;
190 substepIndex++;
191 return currentSubstepSize;
192 break;
193 }
194 case FirstHalfStep: {
195 return 0.5 * currentSubstepSize;
196 break;
197 }
198 case SecondHalfStep: {
199 return 0.5 * currentSubstepSize;
200 break;
201 }
202 }
203 return 0;
204 }

228 template < size_t n, size_t nState >
229 bool AdaptiveSubstepperExplicit< n, nState >::discardSubstep()
230 {
231 passedSubsteps = 0;
232 switch ( currentState ) {
233 case FullStep: {
234 currentSubstepSize *= scaleDownFactor; // we use the scale factor only here
235 break;
236 }
237 // these cases should actually never happen, as the full step has already converged!
238 case FirstHalfStep:
239 MarmotJournal::warningToMSG( "UMAT: warning, 1th half sub step has not converged after already "
240 "converged full step" );
241 return acceptSubstepWithFullStepOnly();
242
243 case SecondHalfStep:
244 MarmotJournal::warningToMSG( "UMAT: warning, 2th half sub step has not converged after already "
245 "converged full step" );
246 return acceptSubstepWithFullStepOnly();
247 }
248
249 currentState = FullStep;
250
251 if ( currentSubstepSize < minimumStepSize )
252 return MarmotJournal::warningToMSG( "UMAT: Substepper: Minimal stepzsize reached" );
253 else
254 return true;
255 }

257 template < size_t n, size_t nState >
258 bool AdaptiveSubstepperExplicit< n, nState >::repeatSubstep( double factorNew )
259 {
260 currentState = FullStep;
261 passedSubsteps = 0;
262
263 currentSubstepSize *= factorNew; // we use the scale factor only here
264
265 if ( currentSubstepSize < minimumStepSize )
266 return MarmotJournal::warningToMSG( "UMAT: Substepper: Minimal stepzsize reached" );
267 else
268 return true;
269 }

271 template < size_t n, size_t nState >
272 bool AdaptiveSubstepperExplicit< n, nState >::finishSubstep( const Marmot::Vector6d& resultStress,
273 const TangentSizedMatrix& dXdY,
274 const TangentSizedMatrix& dYdXOld,
275 const IntegrationStateVector& stateVars )
276 {
277 if ( currentState == FullStep ) {
278 stressProgressFullTemp = resultStress;
279 stateProgressFullTemp = stateVars;
280 consistentTangentProgressFullTemp = consistentTangentProgress;
281 consistentTangentProgressFullTemp.applyOnTheLeft( IXX() - dYdXOld );
282 consistentTangentProgressFullTemp += currentSubstepSize * IX6() * Cel;
283 consistentTangentProgressFullTemp.applyOnTheLeft( dXdY );
284 currentState = FirstHalfStep;
285 return true;
286 }
287 else if ( currentState == FirstHalfStep ) {
288
289 stressProgressHalfTemp = resultStress;
290 stateProgressHalfTemp = stateVars;
291 consistentTangentProgressHalfTemp = consistentTangentProgress;
292 consistentTangentProgressHalfTemp.applyOnTheLeft( IXX() - dYdXOld );
293 consistentTangentProgressHalfTemp += 0.5 * currentSubstepSize * IX6() * Cel;
294 consistentTangentProgressHalfTemp.applyOnTheLeft( dXdY );
295 currentState = SecondHalfStep;
296 return true;
297 }
298
299 else if ( currentState == SecondHalfStep ) {
300 // error Estimation
301
302 currentState = FullStep;
303 const double error = ( resultStress - stressProgressFullTemp ).norm();
304 const double errorRatio = error / integrationErrorTolerance;
305 double scaleFactor = 1.0;
306 if ( errorRatio > 1e-10 )
307 scaleFactor = 0.9 * std::sqrt( 1. / errorRatio );
308
309 // saturations
310 if ( scaleFactor < 0.1 )
311 scaleFactor = 0.1;
312 if ( scaleFactor * currentSubstepSize < minimumStepSize )
313 scaleFactor = minimumStepSize / currentSubstepSize;
314 if ( scaleFactor > maxScaleUpFactor )
315 scaleFactor = maxScaleUpFactor;
316 if ( scaleFactor > 10 )
317 scaleFactor = 10;
318
319 // Error large than tolerance?
320 if ( error > integrationErrorTolerance ) {
321 discardedDueToError++;
322 passedSubsteps = 0;
323 if ( errorRatio < 2 ) {
324 return splitCurrentSubstep();
325 }
326 else {
327 return repeatSubstep( scaleFactor );
328 }
329 // return splitCurrentSubstep(); }
330 }
331 else {
332
333 stressProgressHalfTemp = resultStress;
334 stateProgressHalfTemp = stateVars;
335 consistentTangentProgressHalfTemp.applyOnTheLeft( IXX() - dYdXOld );
336 consistentTangentProgressHalfTemp += 0.5 * currentSubstepSize * IX6() * Cel;
337 consistentTangentProgressHalfTemp.applyOnTheLeft( dXdY );
338
339 consistentTangentProgress = 2 * consistentTangentProgressHalfTemp - consistentTangentProgressFullTemp;
340 stressProgress = 2 * stressProgressHalfTemp - stressProgressFullTemp;
341 stateProgress = 2 * stateProgressHalfTemp - stateProgressFullTemp;
342 currentProgress += currentSubstepSize;
343
344 passedSubsteps++;
345 currentSubstepSize *= scaleFactor;
346
347 return true;
348 }
349 }
350 return false;
351 }

353 template < size_t n, size_t nState >
354 void AdaptiveSubstepperExplicit< n, nState >::finishElasticSubstep( const Marmot::Vector6d& newStress )
355 {
356 switch ( currentState ) {
357 case FullStep: {
358 // this means, that the complete current cycle is already successfull,
359 // as the two half steps must also be elastic!
360 consistentTangentProgress += currentSubstepSize * IX6() * Cel;
361 stressProgress = newStress;
362 // no need for two half steps if full step was already elastic
363 currentProgress += currentSubstepSize;
364 currentState = FullStep;
365 passedSubsteps++;
366 break;
367 }
368
369 case FirstHalfStep: {
370 consistentTangentProgressHalfTemp = consistentTangentProgress;
371 consistentTangentProgressHalfTemp += 0.5 * currentSubstepSize * IX6() * Cel;
372 stressProgressHalfTemp = newStress;
373 currentState = SecondHalfStep;
374 break;
375 }
376 case SecondHalfStep: {
377 acceptSubstepWithFullStepOnly();
378 break;
379 }
380 }
381 }

394 bool AdaptiveSubstepperExplicit< n, nState >::acceptSubstepWithFullStepOnly()
395 {
396 consistentTangentProgress = consistentTangentProgressFullTemp;

406 template < size_t n, size_t nState >
407 bool AdaptiveSubstepperExplicit< n, nState >::splitCurrentSubstep()
408 {
409 if ( currentSubstepSize < 2 * minimumStepSize ) {
410 if ( ignoreErrorToleranceOnMinimumStepSize )
411 return acceptSubstepWithFullStepOnly();
412 else
413 return false;
414 }
415
416 consistentTangentProgressFullTemp = consistentTangentProgressHalfTemp;
417 stressProgressFullTemp = stressProgressHalfTemp;
418 stateProgressFullTemp = stateProgressHalfTemp;
419 currentSubstepSize *= 0.5;
420 currentState = FirstHalfStep;
421
422 return true;
423 }
