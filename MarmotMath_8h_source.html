61  template <int ySize, typename functionType, typename... Args>
62  Eigen::Matrix<double, ySize, 1> semiImplicitEuler( Eigen::Matrix<double, ySize, 1> yN, const double dt, functionType fRate, Args&&... fRateArgs )
63  {
64  Eigen::Matrix<double, ySize, ySize> fS = Eigen::Matrix<double, ySize, ySize>::Zero();
65  Eigen::Matrix<double, ySize, ySize> Iy = Eigen::Matrix<double, ySize, ySize>::Identity();
66 
67  Eigen::VectorXd leftX( ySize );
68  Eigen::VectorXd rightX( ySize );
69 
70  for ( size_t i = 0; i < ySize; i++ ) {
71  double volatile h = std::max( 1.0, std::abs( yN( i ) ) ) * Constants::cubicRootEps();
72  leftX = yN;
73  leftX( i ) -= h;
74  rightX = yN;
75  rightX( i ) += h;
76  fS.col( i ) = 1. / ( 2. * h ) * ( fRate( rightX, fRateArgs... ) - fRate( leftX, fRateArgs... ) );
77  }
78 
79  return yN + ( Iy - dt * fS ).inverse() * dt * fRate( yN, fRateArgs... );
80  }

84  template <int nRows, int nCols, typename functionType>
85  Eigen::Matrix<double, nRows, nCols> centralDiff( functionType f, const Eigen::Matrix<double, nCols, 1>& X )
86  {
87  Eigen::Matrix<double, nRows, nCols> dXdY = Eigen::Matrix<double, nRows, nCols>::Zero();
88 
89  Eigen::VectorXd leftX( nCols );
90  Eigen::VectorXd rightX( nCols );
91 
92  for ( size_t i = 0; i < nCols; i++ ) {
93  double volatile h = std::max( 1.0, std::abs( X( i ) ) ) * Constants::cubicRootEps();
94  leftX = X;
95  leftX( i ) -= h;
96  rightX = X;
97  rightX( i ) += h;

105  yType explicitEulerRichardson( yType yN, const double dt, functionType fRate, Args&&... fRateArgs )
106  {

115  template <int ySize, typename functionType, typename... Args>
116  std::tuple<Eigen::Matrix<double, ySize, 1>, double> explicitEulerRichardsonWithErrorEstimator( Eigen::Matrix<double, ySize, 1> yN, const double dt, const double TOL, functionType fRate, Args&&... fRateArgs )
117  {
118 
119  typedef Eigen::Matrix<double, ySize, 1> ySized;
120  ySized fN = fRate( yN, fRateArgs... );
121  ySized u = yN + fN * dt;
122  ySized v = yN + fN * dt / 2.;
123  ySized w = v + fRate( v, fRateArgs... ) * dt / 2.;
124  ySized yNew = 2. * w - u;
125 
126  // error estimator
127  const double AERR = 1.0;
128  const double aI = AERR/TOL;
129  const double rI = 1.0;
130  double scaling = 0;
131  ySized ESTVec = ySized::Zero();
132 
133  for (int i =0; i < ySize; i++){
134  scaling = aI + rI * std::max( abs(yNew(i)), abs(yN(i)) );
135  ESTVec(i) = abs(w(i) - u(i) ) / abs( scaling );
136  }
137

139  const double tauNew = dt * std::min(2., std::max (0.2, 0.9 * std::sqrt(TOL/EST)));
