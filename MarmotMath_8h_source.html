117 template < int ySize, typename functionType, typename... Args >
118 Eigen::Matrix< double, ySize, 1 > semiImplicitEuler( Eigen::Matrix< double, ySize, 1 > yN,
119 const double dt,
120 functionType fRate,
121 Args&&... fRateArgs )
122 {
123 Eigen::Matrix< double, ySize, ySize > fS = Eigen::Matrix< double, ySize, ySize >::Zero();
124 Eigen::Matrix< double, ySize, ySize > Iy = Eigen::Matrix< double, ySize, ySize >::Identity();
125
126 Eigen::VectorXd leftX( ySize );
127 Eigen::VectorXd rightX( ySize );
128
129 for ( size_t i = 0; i < ySize; i++ ) {
130 double volatile h = std::max( 1.0, std::abs( yN( i ) ) ) * Constants::cubicRootEps();
131 leftX = yN;
132 leftX( i ) -= h;
133 rightX = yN;
134 rightX( i ) += h;
135 fS.col( i ) = 1. / ( 2. * h ) * ( fRate( rightX, fRateArgs... ) - fRate( leftX, fRateArgs... ) );
136 }
137
138 return yN + ( Iy - dt * fS ).inverse() * dt * fRate( yN, fRateArgs... );
139 }

144 template < int nRows, int nCols, typename functionType >
145 Eigen::Matrix< double, nRows, nCols > centralDiff( functionType f, const Eigen::Matrix< double, nCols, 1 >& X )
146 {
147 Eigen::Matrix< double, nRows, nCols > dXdY = Eigen::Matrix< double, nRows, nCols >::Zero();
148
149 Eigen::VectorXd leftX( nCols );
150 Eigen::VectorXd rightX( nCols );
151
152 for ( size_t i = 0; i < nCols; i++ ) {
153 double volatile h = std::max( 1.0, std::abs( X( i ) ) ) * Constants::cubicRootEps();
154 leftX = X;
155 leftX( i ) -= h;
156 rightX = X;
157 rightX( i ) += h;
158 dXdY.col( i ) = 1. / ( 2. * h ) * ( f( rightX ) - f( leftX ) );
159 }
160
161 return dXdY;
162 }

168 template < typename functionType, typename yType, typename... Args >
169 yType explicitEulerRichardson( yType yN, const double dt, functionType fRate, Args&&... fRateArgs )
170 {
171 yType fN = fRate( yN, fRateArgs... );
172 yType u = yN + fN * dt;
173 yType v = yN + fN * dt / 2.;
174 yType w = v + fRate( v, fRateArgs... ) * dt / 2.;
175
176 return 2. * w - u;
177 }

183 template < int ySize, typename functionType, typename... Args >
184 std::tuple< Eigen::Matrix< double, ySize, 1 >, double > explicitEulerRichardsonWithErrorEstimator(
185 Eigen::Matrix< double, ySize, 1 > yN,
186 const double dt,
187 const double TOL,
188 functionType fRate,
189 Args&&... fRateArgs )
190 {
191
192 typedef Eigen::Matrix< double, ySize, 1 > ySized;
193 ySized fN = fRate( yN, fRateArgs... );
194 ySized u = yN + fN * dt;
195 ySized v = yN + fN * dt / 2.;
196 ySized w = v + fRate( v, fRateArgs... ) * dt / 2.;
197 ySized yNew = 2. * w - u;
198
199 // error estimator
200 const double AERR = 1.0;
201 const double aI = AERR / TOL;
202 const double rI = 1.0;
203 double scaling = 0;
204 ySized ESTVec = ySized::Zero();
205
206 for ( int i = 0; i < ySize; i++ ) {
207 scaling = aI + rI * std::max( abs( yNew( i ) ), abs( yN( i ) ) );
208 ESTVec( i ) = abs( w( i ) - u( i ) ) / abs( scaling );
209 }
210
211 const double EST = ESTVec.maxCoeff();
212 const double tauNew = dt * std::min( 2., std::max( 0.2, 0.9 * std::sqrt( TOL / EST ) ) );
213
214 return std::make_tuple( yNew, tauNew );
215 }
