113  template < int ySize, typename functionType, typename... Args >
114  Eigen::Matrix< double, ySize, 1 > semiImplicitEuler( Eigen::Matrix< double, ySize, 1 > yN,
115  const double dt,
116  functionType fRate,
117  Args&&... fRateArgs )
118  {
119  Eigen::Matrix< double, ySize, ySize > fS = Eigen::Matrix< double, ySize, ySize >::Zero();
120  Eigen::Matrix< double, ySize, ySize > Iy = Eigen::Matrix< double, ySize, ySize >::Identity();
121 
122  Eigen::VectorXd leftX( ySize );
123  Eigen::VectorXd rightX( ySize );
124 
125  for ( size_t i = 0; i < ySize; i++ ) {
126  double volatile h = std::max( 1.0, std::abs( yN( i ) ) ) * Constants::cubicRootEps();
127  leftX = yN;
128  leftX( i ) -= h;
129  rightX = yN;
130  rightX( i ) += h;
131  fS.col( i ) = 1. / ( 2. * h ) * ( fRate( rightX, fRateArgs... ) - fRate( leftX, fRateArgs... ) );
132  }
133 
134  return yN + ( Iy - dt * fS ).inverse() * dt * fRate( yN, fRateArgs... );
135  }

179  template < int ySize, typename functionType, typename... Args >
180  std::tuple< Eigen::Matrix< double, ySize, 1 >, double > explicitEulerRichardsonWithErrorEstimator(
181  Eigen::Matrix< double, ySize, 1 > yN,
182  const double dt,
183  const double TOL,
184  functionType fRate,
185  Args&&... fRateArgs )
186  {
187 
188  typedef Eigen::Matrix< double, ySize, 1 > ySized;
189  ySized fN = fRate( yN, fRateArgs... );
190  ySized u = yN + fN * dt;
191  ySized v = yN + fN * dt / 2.;
192  ySized w = v + fRate( v, fRateArgs... ) * dt / 2.;
193  ySized yNew = 2. * w - u;
194 
195  // error estimator
196  const double AERR = 1.0;
197  const double aI = AERR / TOL;
198  const double rI = 1.0;
199  double scaling = 0;
200  ySized ESTVec = ySized::Zero();
201 
202  for ( int i = 0; i < ySize; i++ ) {
203  scaling = aI + rI * std::max( abs( yNew( i ) ), abs( yN( i ) ) );
204  ESTVec( i ) = abs( w( i ) - u( i ) ) / abs( scaling );
205  }
206 
207  const double EST = ESTVec.maxCoeff();
208  const double tauNew = dt * std::min( 2., std::max( 0.2, 0.9 * std::sqrt( TOL / EST ) ) );
209 
210  return std::make_tuple( yNew, tauNew );
211  }
