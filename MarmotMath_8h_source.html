166  template < int ySize, typename functionType, typename... Args >
167  Eigen::Matrix< double, ySize, 1 > semiImplicitEuler( Eigen::Matrix< double, ySize, 1 > yN,
168  const double dt,
169  functionType fRate,
170  Args&&... fRateArgs )
171  {
172  Eigen::Matrix< double, ySize, ySize > fS = Eigen::Matrix< double, ySize, ySize >::Zero();
173  Eigen::Matrix< double, ySize, ySize > Iy = Eigen::Matrix< double, ySize, ySize >::Identity();
174 
175  Eigen::VectorXd leftX( ySize );
176  Eigen::VectorXd rightX( ySize );
177 
178  for ( size_t i = 0; i < ySize; i++ ) {
179  double volatile h = std::max( 1.0, std::abs( yN( i ) ) ) * Constants::cubicRootEps();
180  leftX = yN;
181  leftX( i ) -= h;
182  rightX = yN;
183  rightX( i ) += h;
184  fS.col( i ) = 1. / ( 2. * h ) * ( fRate( rightX, fRateArgs... ) - fRate( leftX, fRateArgs... ) );
185  }
186 
187  return yN + ( Iy - dt * fS ).inverse() * dt * fRate( yN, fRateArgs... );
188  }

195  yType explicitEulerRichardson( yType yN, const double dt, functionType fRate, Args&&... fRateArgs )
196  {
197  yType fN = fRate( yN, fRateArgs... );

209  template < int ySize, typename functionType, typename... Args >
210  std::tuple< Eigen::Matrix< double, ySize, 1 >, double > explicitEulerRichardsonWithErrorEstimator(
211  Eigen::Matrix< double, ySize, 1 > yN,
212  const double dt,
213  const double TOL,
214  functionType fRate,
215  Args&&... fRateArgs )
216  {
217 
218  typedef Eigen::Matrix< double, ySize, 1 > ySized;
219  ySized fN = fRate( yN, fRateArgs... );
220  ySized u = yN + fN * dt;
221  ySized v = yN + fN * dt / 2.;
222  ySized w = v + fRate( v, fRateArgs... ) * dt / 2.;
223  ySized yNew = 2. * w - u;
224 
225  // error estimator
226  const double AERR = 1.0;
227  const double aI = AERR / TOL;
228  const double rI = 1.0;
229  double scaling = 0;
230  ySized ESTVec = ySized::Zero();
231 
232  for ( int i = 0; i < ySize; i++ ) {
233  scaling = aI + rI * std::max( abs( yNew( i ) ), abs( yN( i ) ) );
234  ESTVec( i ) = abs( w( i ) - u( i ) ) / abs( scaling );
235  }
236 
237  const double EST = ESTVec.maxCoeff();
238  const double tauNew = dt * std::min( 2., std::max( 0.2, 0.9 * std::sqrt( TOL / EST ) ) );
239 
240  return std::make_tuple( yNew, tauNew );
241  }
