78 template < int ySize, typename functionType, typename... Args >
79 Eigen::Matrix< double, ySize, 1 > semiImplicitEuler( Eigen::Matrix< double, ySize, 1 > yN,
80 const double dt,
81 functionType fRate,
82 Args&&... fRateArgs )
83 {
84 Eigen::Matrix< double, ySize, ySize > fS = Eigen::Matrix< double, ySize, ySize >::Zero();
85 Eigen::Matrix< double, ySize, ySize > Iy = Eigen::Matrix< double, ySize, ySize >::Identity();
86
87 Eigen::VectorXd leftX( ySize );
88 Eigen::VectorXd rightX( ySize );
89
90 for ( size_t i = 0; i < ySize; i++ ) {
91 double volatile h = std::max( 1.0, std::abs( yN( i ) ) ) * Constants::cubicRootEps();
92 leftX = yN;
93 leftX( i ) -= h;
94 rightX = yN;
95 rightX( i ) += h;
96 fS.col( i ) = 1. / ( 2. * h ) * ( fRate( rightX, fRateArgs... ) - fRate( leftX, fRateArgs... ) );
97 }
98
99 return yN + ( Iy - dt * fS ).inverse() * dt * fRate( yN, fRateArgs... );
100 }

102 // returns central numerical differentiation of a vector-valued function f(x) with respect to vector x
103 // --> use std::bind to create function f(x) from function with multiple arguments
104 template < int nRows, int nCols, typename functionType >
105 Eigen::Matrix< double, nRows, nCols > centralDiff( functionType f, const Eigen::Matrix< double, nCols, 1 >& X )
106 {
107 Eigen::Matrix< double, nRows, nCols > dXdY = Eigen::Matrix< double, nRows, nCols >::Zero();
108
109 Eigen::VectorXd leftX( nCols );
110 Eigen::VectorXd rightX( nCols );
111
112 for ( size_t i = 0; i < nCols; i++ ) {
113 double volatile h = std::max( 1.0, std::abs( X( i ) ) ) * Constants::cubicRootEps();
114 leftX = X;
115 leftX( i ) -= h;
116 rightX = X;
117 rightX( i ) += h;
118 dXdY.col( i ) = 1. / ( 2. * h ) * ( f( rightX ) - f( leftX ) );
119 }
120
121 return dXdY;
122 }

124 template < typename functionType, typename yType, typename... Args >
125 yType explicitEulerRichardson( yType yN, const double dt, functionType fRate, Args&&... fRateArgs )
126 {
127 yType fN = fRate( yN, fRateArgs... );
128 yType u = yN + fN * dt;
129 yType v = yN + fN * dt / 2.;
130 yType w = v + fRate( v, fRateArgs... ) * dt / 2.;
131
132 return 2. * w - u;
133 }

135 template < int ySize, typename functionType, typename... Args >
136 std::tuple< Eigen::Matrix< double, ySize, 1 >, double > explicitEulerRichardsonWithErrorEstimator(
137 Eigen::Matrix< double, ySize, 1 > yN,
138 const double dt,
139 const double TOL,
140 functionType fRate,
141 Args&&... fRateArgs )
142 {
143
144 typedef Eigen::Matrix< double, ySize, 1 > ySized;
145 ySized fN = fRate( yN, fRateArgs... );
146 ySized u = yN + fN * dt;
147 ySized v = yN + fN * dt / 2.;
148 ySized w = v + fRate( v, fRateArgs... ) * dt / 2.;
149 ySized yNew = 2. * w - u;
150
151 // error estimator
152 const double AERR = 1.0;
153 const double aI = AERR / TOL;
154 const double rI = 1.0;
155 double scaling = 0;
156 ySized ESTVec = ySized::Zero();
157
158 for ( int i = 0; i < ySize; i++ ) {
159 scaling = aI + rI * std::max( abs( yNew( i ) ), abs( yN( i ) ) );
160 ESTVec( i ) = abs( w( i ) - u( i ) ) / abs( scaling );
161 }
162
163 const double EST = ESTVec.maxCoeff();
164 const double tauNew = dt * std::min( 2., std::max( 0.2, 0.9 * std::sqrt( TOL / EST ) ) );
165
166 return std::make_tuple( yNew, tauNew );
167 }
