103  template <int n>
104  double PerezFougetSubstepper<n>::getNextSubstep()
105  {
106  if ( passedSubsteps >= nPassesToIncrease )
107  currentSubstepSize *= scaleUpFactor;
108 
109  const double remainingProgress = 1.0 - currentProgress;
110  if ( remainingProgress < currentSubstepSize )
111  currentSubstepSize = remainingProgress;
112 
113  passedSubsteps++;
114  currentProgress += currentSubstepSize;
115 
116  return currentSubstepSize;
117  }
118 
119  template <int n>
120  bool PerezFougetSubstepper<n>::decreaseSubstepSize()
121  {
122  currentProgress -= currentSubstepSize;
123  passedSubsteps = 0;
124 
125  currentSubstepSize *= scaleDownFactor;
126 
127  if ( currentSubstepSize < minimumStepSize )
128  return MarmotJournal::warningToMSG( "UMAT: Substepper: Minimal stepzsize reached" );
129  else
130  return MarmotJournal::notificationToMSG( "UMAT: Substepper: Decreasing stepsize" );
131  }
