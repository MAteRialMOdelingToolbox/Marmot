34template < int nDim, int nNodes >
35class MarmotGeometryElement {
36 /* This is the Geometry Base element, which serves as a base for all MarmotElements.
37 * It corresponds to the GeometryElement in mpFEM,
38 * although this as a static templated version.
39 *
40 * MarmotElements (corresponding do DofElements in mpFEM) can inherit from this element,
41 * and access shape functions, derivatives and B Operator
42 *
43 * The element automatically determines its shape by the given nDimension and number of nodes
44 * */
45
46public:
47 /*Typedefs*/
48 static constexpr int VoigtSize = ( ( ( nDim * nDim ) + nDim ) / 2 );
49
50 typedef Eigen::Matrix< double, nDim, 1 > XiSized;
51 typedef Eigen::Matrix< double, nDim * nNodes, 1 > CoordinateVector;
52 typedef Eigen::Matrix< double, nDim, nDim > JacobianSized;
53 typedef Eigen::Matrix< double, 1, nNodes > NSized;
54 typedef Eigen::Matrix< double, nDim, nNodes * nDim > NBSized;
55 typedef Eigen::Matrix< double, nDim, nNodes > dNdXiSized;
56 typedef Eigen::Matrix< double, VoigtSize, nNodes * nDim > BSized;
57
58 /*Properties*/
59 Eigen::Map< const CoordinateVector > coordinates;
60 const Marmot::FiniteElement::ElementShapes shape;
61
62 /*Methods*/
63 MarmotGeometryElement()
64 : coordinates( nullptr ), shape( Marmot::FiniteElement::getElementShapeByMetric( nDim, nNodes ) ){};
65
66 std::string getElementShape()
67 {
68 using namespace Marmot::FiniteElement;
69 static std::map< ElementShapes, std::string > shapes = { { Bar2, "bar2" },
70 { Quad4, "quad4" },
71 { Quad8, "quad8" },
72 { Tetra4, "tetra4" },
73 { Tetra10, "tetra10" },
74 { Hexa8, "hexa8" },
75 { Hexa20, "hexa20" } };
76
77 return shapes[this->shape];
78 }
79
80 void initializeYourself( const double* coords )
81 {
82 new ( &coordinates ) Eigen::Map< const CoordinateVector >( coords );
83 }
84
85 /*Please specialize these functions for each element individially
86 *.cpp file.
87 *Fully specialized templates are precompiled in marmotMechanics (rather than the unspecialized and
88 *partially specialized templates)
89 * */
90 NSized N( const XiSized& xi );
91 dNdXiSized dNdXi( const XiSized& xi );
92 BSized B( const dNdXiSized& dNdX );
93 BSized BGreen( const dNdXiSized& dNdX, const JacobianSized& F );
94
95 /*These functions are equal for each element and independent of node number and nDimension*/
96 NBSized NB( const NSized& N ) { return Marmot::FiniteElement::NB< nDim, nNodes >( N ); }
97
98 JacobianSized Jacobian( const dNdXiSized& dNdXi )
99 {
100 return Marmot::FiniteElement::Jacobian< nDim, nNodes >( dNdXi, coordinates );
101 }
102
103 dNdXiSized dNdX( const dNdXiSized& dNdXi, const JacobianSized& JacobianInverse )
104 {
105 return ( dNdXi.transpose() * JacobianInverse ).transpose();
106 }
107
108 JacobianSized F( const dNdXiSized& dNdX, const CoordinateVector& Q )
109 {
110 return Marmot::FiniteElement::Jacobian< nDim, nNodes >( dNdX, Q ) + JacobianSized::Identity();
111 }
112};
