35template < int nDim, int nNodes >
36class MarmotGeometryElement {
37 /* This is the Geometry Base element, which serves as a base for all MarmotElements.
38 * It corresponds to the GeometryElement in mpFEM,
39 * although this as a static templated version.
40 *
41 * MarmotElements (corresponding do DofElements in mpFEM) can inherit from this element,
42 * and access shape functions, derivatives and B Operator
43 *
44 * The element automatically determines its shape by the given nDimension and number of nodes
45 * */
46
47public:
48 /*Typedefs*/
49 /* static constexpr VoigtSize voigtSize = ( ( ( nDim * nDim ) + nDim ) / 2 ); */
50 static constexpr Marmot::ContinuumMechanics::VoigtNotation::VoigtSize voigtSize = Marmot::ContinuumMechanics::VoigtNotation::voigtSizeFromDimension ( nDim );
51
52 typedef Eigen::Matrix< double, nDim, 1 > XiSized;
53 typedef Eigen::Matrix< double, nDim * nNodes, 1 > CoordinateVector;
54 typedef Eigen::Matrix< double, nDim, nDim > JacobianSized;
55 typedef Eigen::Matrix< double, 1, nNodes > NSized;
56 typedef Eigen::Matrix< double, nDim, nNodes * nDim > NBSized;
57 typedef Eigen::Matrix< double, nDim, nNodes > dNdXiSized;
58 typedef Eigen::Matrix< double, voigtSize, nNodes * nDim > BSized;
59
60 /*Properties*/
61 Eigen::Map< const CoordinateVector > coordinates;
62 const Marmot::FiniteElement::ElementShapes shape;
63
64 /*Methods*/
65 MarmotGeometryElement()
66 : coordinates( nullptr ), shape( Marmot::FiniteElement::getElementShapeByMetric( nDim, nNodes ) ){};
67
68 std::string getElementShape()
69 {
70 using namespace Marmot::FiniteElement;
71 static std::map< ElementShapes, std::string > shapes = { { Bar2, "bar2" },
72 { Quad4, "quad4" },
73 { Quad8, "quad8" },
74 { Tetra4, "tetra4" },
75 { Tetra10, "tetra10" },
76 { Hexa8, "hexa8" },
77 { Hexa20, "hexa20" } };
78
79 return shapes[this->shape];
80 }
81
82 void initializeYourself( const double* coords )
83 {
84 new ( &coordinates ) Eigen::Map< const CoordinateVector >( coords );
85 }
86
87 /*Please specialize these functions for each element individially
88 *.cpp file.
89 *Fully specialized templates are precompiled in marmotMechanics (rather than the unspecialized and
90 *partially specialized templates)
91 * */
92 NSized N( const XiSized& xi );
93 dNdXiSized dNdXi( const XiSized& xi );
94 BSized B( const dNdXiSized& dNdX );
95 BSized BGreen( const dNdXiSized& dNdX, const JacobianSized& F );
96
97 /*These functions are equal for each element and independent of node number and nDimension*/
98 NBSized NB( const NSized& N ) { return Marmot::FiniteElement::NB< nDim, nNodes >( N ); }
99
100 JacobianSized Jacobian( const dNdXiSized& dNdXi )
101 {
102 return Marmot::FiniteElement::Jacobian< nDim, nNodes >( dNdXi, coordinates );
103 }
104
105 dNdXiSized dNdX( const dNdXiSized& dNdXi, const JacobianSized& JacobianInverse )
106 {
107 return ( dNdXi.transpose() * JacobianInverse ).transpose();
108 }
109
110 JacobianSized F( const dNdXiSized& dNdX, const CoordinateVector& Q )
111 {
112 return Marmot::FiniteElement::Jacobian< nDim, nNodes >( dNdX, Q ) + JacobianSized::Identity();
113 }
114};
