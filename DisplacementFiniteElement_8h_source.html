222 template < int nDim, int nNodes >
223 void DisplacementFiniteElement< nDim, nNodes >::assignStateVars( double* stateVars, int nStateVars )
224 {
225 /* we provide as many statevars to the material as we can (some materials store addition debugging infos if
226 * possible alternatively: int nStateVarsMaterial =
227 * quadraturePoints[0].material->getNumberOfRequiredStateVars();
228 * */
229 int nStateVarsMaterial = ( nStateVars / quadraturePoints.size() ) - QuadraturePoint::nRequiredStateVars;
230
231 for ( size_t i = 0; i < quadraturePoints.size(); i++ ) {
232
233 QuadraturePoint& qp = quadraturePoints[i];
234 qp.material->assignStateVars( stateVars + i * ( nStateVarsMaterial + QuadraturePoint::nRequiredStateVars ),
235 nStateVarsMaterial );
236
237 // assign stress, strain state vars using the 'placement new' operator
238 new ( &qp.stress )
239 mVector6d( stateVars + nStateVarsMaterial + i * ( nStateVarsMaterial + QuadraturePoint::nRequiredStateVars ),
240 6 );
241
242 new ( &qp.strain ) mVector6d( stateVars + nStateVarsMaterial +
243 i * ( nStateVarsMaterial + QuadraturePoint::nRequiredStateVars ) + 6,
244 6 );
245 }
246 }

268 template < int nDim, int nNodes >
269 void DisplacementFiniteElement< nDim, nNodes >::initializeYourself( const double* coordinates )
270 {
271 ParentGeometryElement::initializeYourself( coordinates );
272
273 for ( QuadraturePoint& qp : quadraturePoints ) {
274
275 qp.geometry = std::make_unique< typename QuadraturePoint::Geometry >();
276
277 qp.geometry->dNdXi = this->dNdXi( qp.xi );
278 qp.geometry->J = this->Jacobian( qp.geometry->dNdXi );
279 qp.geometry->JInv = qp.geometry->J.inverse();
280 qp.geometry->detJ = qp.geometry->J.determinant();
281 qp.geometry->dNdX = this->dNdX( qp.geometry->dNdXi, qp.geometry->JInv );
282 qp.geometry->B = this->B( qp.geometry->dNdX );
283
284 if ( sectionType == SectionType::Solid ) {
285
286 qp.geometry->intVol = qp.weight * qp.geometry->detJ;
287 qp.material->setCharacteristicElementLength( std::cbrt( 8 * qp.geometry->detJ ) );
288 }
289
290 else if ( sectionType == SectionType::PlaneStrain || sectionType == SectionType::PlaneStress ) {
291
292 const double& thickness = elementProperties[0];
293 qp.geometry->intVol = qp.weight * qp.geometry->detJ * thickness;
294 qp.material->setCharacteristicElementLength( std::sqrt( 4 * qp.geometry->detJ ) );
295 }
296
297 else if ( sectionType == SectionType::UniaxialStress ) {
298
299 const double& crossSection = elementProperties[0];
300 qp.geometry->intVol = qp.weight * qp.geometry->detJ * crossSection;
301 qp.material->setCharacteristicElementLength( 2 * qp.geometry->detJ );
302 }
303 }

306 template < int nDim, int nNodes >
307 void DisplacementFiniteElement< nDim, nNodes >::computeYourself( const double* QTotal_,
308 const double* dQ_,
309 double* Pe_,
310 double* Ke_,
311 const double* time,
312 double dT,
313 double& pNewDT )
314 {
315 using namespace Marmot;
316
317 Map< const RhsSized > QTotal( QTotal_ );
318 Map< const RhsSized > dQ( dQ_ );
319 Map< KeSizedMatrix > Ke( Ke_ );
320 Map< RhsSized > Pe( Pe_ );
321
322 Voigt S, dE;
323 CSized C;
324
325 for ( QuadraturePoint& quadraturePoint : quadraturePoints ) {
326
327 const BSized& B = quadraturePoint.geometry->B;
328
329 dE = B * dQ;
330
331 if constexpr ( nDim == 1 ) {
332 Vector6d dE6 = ( Vector6d() << dE, 0, 0, 0, 0, 0 ).finished();
333 Matrix6d C66;
334 quadraturePoint.material
335 ->computeUniaxialStress( quadraturePoint.stress.data(), C66.data(), dE6.data(), time, dT, pNewDT );
336
337 C << ContinuumMechanics::UniaxialStress::getUniaxialStressTangent( C66 );
338 S( 0 ) = quadraturePoint.stress( 0 );
339 }
340
341 else if constexpr ( nDim == 2 ) {
342
343 Vector6d dE6 = ContinuumMechanics::VoigtNotation::planeVoigtToVoigt( dE );
344 Matrix6d C66;
345
346 if ( sectionType == SectionType::PlaneStress ) {
347
348 quadraturePoint.material
349 ->computePlaneStress( quadraturePoint.stress.data(), C66.data(), dE6.data(), time, dT, pNewDT );
350
351 C = ContinuumMechanics::PlaneStress::getPlaneStressTangent( C66 );
352 }
353
354 else if ( sectionType == SectionType::PlaneStrain ) {
355
356 quadraturePoint.material
357 ->computeStress( quadraturePoint.stress.data(), C66.data(), dE6.data(), time, dT, pNewDT );
358
359 C = ContinuumMechanics::PlaneStrain::getPlaneStrainTangent( C66 );
360 }
361
362 S = ContinuumMechanics::VoigtNotation::voigtToPlaneVoigt( quadraturePoint.stress );
363 quadraturePoint.strain += dE6;
364 }
365
366 else if constexpr ( nDim == 3 ) {
367
368 if ( sectionType == SectionType::Solid ) {
369
370 quadraturePoint.material
371 ->computeStress( quadraturePoint.stress.data(), C.data(), dE.data(), time, dT, pNewDT );
372 }
373
374 S = quadraturePoint.stress;
375 quadraturePoint.strain += dE;
376 }
377
378 if ( pNewDT < 1.0 )
379 return;
380
381 Ke += B.transpose() * C * B * quadraturePoint.geometry->intVol;
382 Pe -= B.transpose() * S * quadraturePoint.geometry->intVol;
383 }
384 }

386 template < int nDim, int nNodes >
387 void DisplacementFiniteElement< nDim, nNodes >::setInitialConditions( StateTypes state, const double* values )
388 {
389 switch ( state ) {
390 case MarmotElement::GeostaticStress: {
391 if constexpr ( nDim == 3 ) {
392 for ( QuadraturePoint& quadraturePoint : quadraturePoints ) {
393
394 XiSized coordAtGauss = this->NB( this->N( quadraturePoint.xi ) ) * this->coordinates;
395
396 const double sigY1 = values[0];
397 const double sigY2 = values[2];
398 const double y1 = values[1];
399 const double y2 = values[3];
400
401 using namespace Math;
402 quadraturePoint.stress( 1 ) = linearInterpolation( coordAtGauss[1], y1, y2, sigY1, sigY2 ); // sigma_y
403 quadraturePoint.stress( 0 ) = values[4] * quadraturePoint.stress( 1 ); // sigma_x
404 quadraturePoint.stress( 2 ) = values[5] * quadraturePoint.stress( 1 );
405 }
406 }
407 break;
408 }
409 case MarmotElement::MarmotMaterialStateVars: {
410 throw std::invalid_argument( "Please use initializeStateVars directly on material" );
411 /* for ( QuadraturePoint& qp : qps ) */
412 /* qp.material->stateVars[static_cast<int>( values[0] )] = values[1]; */
413 /* for ( QuadraturePoint& quadraturePoint : quadraturePoints ) */
414 /* quadraturePoint.material->stateVars[static_cast<int>(values[0])] = values[1]; */
415 /* break; */
416 }
417 default: throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid initial condition" );
418 }
419 /* } */
420 }

422 template < int nDim, int nNodes >
423 void DisplacementFiniteElement< nDim, nNodes >::computeDistributedLoad( MarmotElement::DistributedLoadTypes loadType,
424 double* P,
425 double* K,
426 const int elementFace,
427 const double* load,
428 const double* QTotal,
429 const double* time,
430 double dT )
431 {
432 Map< RhsSized > fU( P );
433
434 switch ( loadType ) {
435
436 case MarmotElement::Pressure: {
437 const double p = load[0];
438
439 FiniteElement::BoundaryElement boundaryEl( this->shape, elementFace, nDim, this->coordinates );
440
441 VectorXd Pk = -p * boundaryEl.computeNormalLoadVector();
442
443 if ( nDim == 2 )
444 Pk *= elementProperties[0]; // thickness
445
446 boundaryEl.assembleIntoParentVector( Pk, fU );
447
448 break;
449 }
450 default: {
451 throw std::invalid_argument( "Invalid Load Type specified" );
452 }
453 }

456 template < int nDim, int nNodes >
457 void DisplacementFiniteElement< nDim, nNodes >::computeBodyForce( double* P_,
458 double* K,
459 const double* load,
460 const double* QTotal,
461 const double* time,
462 double dT )
463 {
464 Map< RhsSized > Pe( P_ );
465 const Map< const Matrix< double, nDim, 1 > > f( load );
466
467 for ( const auto& qp : quadraturePoints )
468 Pe += this->NB( this->N( qp.xi ) ).transpose() * f * qp.geometry->intVol;
469 }
