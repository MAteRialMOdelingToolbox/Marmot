286 template < int nDim, int nNodes >
287 void DisplacementFiniteElement< nDim, nNodes >::initializeYourself( const double* coordinates )
288 {
289 ParentGeometryElement::initializeYourself( coordinates );
290
291 for ( QuadraturePoint& qp : qps ) {
292
293 const dNdXiSized dNdXi = this->dNdXi( qp.xi );
294 const JacobianSized J = this->Jacobian( dNdXi );
295 const JacobianSized JInv = J.inverse();
296 const dNdXiSized dNdX = this->dNdX( dNdXi, JInv );
297 qp.detJ = J.determinant();
298 qp.B = this->B( dNdX );
299
300 if ( sectionType == SectionType::Solid ) {
301
302 qp.J0xW = qp.weight * qp.detJ;
303 qp.material->setCharacteristicElementLength( std::cbrt( 8 * qp.detJ ) );
304 }
305
306 else if ( sectionType == SectionType::PlaneStrain || sectionType == SectionType::PlaneStress ) {
307
308 const double& thickness = elementProperties[0];
309 qp.J0xW = qp.weight * qp.detJ * thickness;
310 qp.material->setCharacteristicElementLength( std::sqrt( 4 * qp.detJ ) );
311 }
312
313 else if ( sectionType == SectionType::UniaxialStress ) {
314
315 const double& crossSection = elementProperties[0];
316 qp.J0xW = qp.weight * qp.detJ * crossSection;
317 qp.material->setCharacteristicElementLength( 2 * qp.detJ );
318 }
319 }
320 }

322 template < int nDim, int nNodes >
323 void DisplacementFiniteElement< nDim, nNodes >::computeYourself( const double* QTotal_,
324 const double* dQ_,
325 double* Pe_,
326 double* Ke_,
327 const double* time,
328 double dT,
329 double& pNewDT )
330 {
331 using namespace Marmot;
332 using namespace ContinuumMechanics::VoigtNotation;
333
334 Map< const RhsSized > QTotal( QTotal_ );
335 Map< const RhsSized > dQ( dQ_ );
336 Map< KeSizedMatrix > Ke( Ke_ );
337 Map< RhsSized > Pe( Pe_ );
338
339 Voigt S, dE;
340 CSized C;
341
342 for ( QuadraturePoint& qp : qps ) {
343
344 const BSized& B = qp.B;
345 dE = B * dQ;
346
347 if constexpr ( nDim == 1 ) {
348
349 S = reduce3DVoigt< ParentGeometryElement::voigtSize >( qp.managedStateVars->stress );
350 qp.material->computeUniaxialStress( S.data(), C.data(), dE.data(), time, dT, pNewDT );
351 qp.managedStateVars->stress = make3DVoigt< ParentGeometryElement::voigtSize >( S );
352 }
353
354 else if constexpr ( nDim == 2 ) {
355
356 if ( sectionType == SectionType::PlaneStress ) {
357
358 S = reduce3DVoigt< ParentGeometryElement::voigtSize >( qp.managedStateVars->stress );
359 qp.material->computePlaneStress( S.data(), C.data(), dE.data(), time, dT, pNewDT );
360 qp.managedStateVars->stress = make3DVoigt< ParentGeometryElement::voigtSize >( S );
361 }
362
363 else if ( sectionType == SectionType::PlaneStrain ) {
364
365 Vector6d dE6 = planeVoigtToVoigt( dE );
366 Matrix6d C66;
367
368 Vector6d S6 = qp.managedStateVars->stress;
369 qp.material->computeStress( S6.data(), C66.data(), dE6.data(), time, dT, pNewDT );
370 qp.managedStateVars->stress = S6;
371
372 S = reduce3DVoigt< ParentGeometryElement::voigtSize >( S6 );
373 C = ContinuumMechanics::PlaneStrain::getPlaneStrainTangent( C66 );
374 }
375 }
376
377 else if constexpr ( nDim == 3 ) {
378 if ( sectionType == SectionType::Solid ) {
379
380 S = qp.managedStateVars->stress;
381 qp.material->computeStress( S.data(), C.data(), dE.data(), time, dT, pNewDT );
382 qp.managedStateVars->stress = S;
383 }
384 }
385
386 qp.managedStateVars->strain += make3DVoigt< ParentGeometryElement::voigtSize >( dE );
387
388 if ( pNewDT < 1.0 )
389 return;
390
391 Ke += B.transpose() * C * B * qp.J0xW;
392 Pe -= B.transpose() * S * qp.J0xW;
393 }
394 }

396 template < int nDim, int nNodes >
397 void DisplacementFiniteElement< nDim, nNodes >::setInitialConditions( StateTypes state, const double* values )
398 {
399 switch ( state ) {
400 case MarmotElement::MarmotMaterialInitialization: {
401 for ( QuadraturePoint& qp : qps ) {
402 qp.material->initializeYourself();
403 }
404 break;
405 }
406 case MarmotElement::GeostaticStress: {
407 for ( QuadraturePoint& qp : qps ) {
408 XiSized coordAtGauss = this->NB( this->N( qp.xi ) ) * this->coordinates;
409
410 const double sigY1 = values[0];
411 const double sigY2 = values[2];
412 const double y1 = values[1];
413 const double y2 = values[3];
414
415 using namespace Math;
416 qp.managedStateVars->stress( 1 ) = linearInterpolation( coordAtGauss[1], y1, y2, sigY1, sigY2 ); // sigma_y
417 qp.managedStateVars->stress( 0 ) = values[4] * qp.managedStateVars->stress( 1 ); // sigma_x
418 qp.managedStateVars->stress( 2 ) = values[5] * qp.managedStateVars->stress( 1 );
419 }
420 break;
421 }
422 case MarmotElement::MarmotMaterialStateVars: {
423 throw std::invalid_argument( "Please use initializeStateVars directly on material" );
424 }
425 default: throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid initial condition" );
426 }
427 /* } */

430 template < int nDim, int nNodes >
431 void DisplacementFiniteElement< nDim, nNodes >::computeDistributedLoad( MarmotElement::DistributedLoadTypes loadType,
432 double* P,
433 double* K,
434 const int elementFace,
435 const double* load,
436 const double* QTotal,
437 const double* time,
438 double dT )
439 {
440 Map< RhsSized > fU( P );
441
442 switch ( loadType ) {
443
444 case MarmotElement::Pressure: {
445 const double p = load[0];
446
447 FiniteElement::BoundaryElement boundaryEl( this->shape, elementFace, nDim, this->coordinates );
448
449 VectorXd Pk = -p * boundaryEl.computeNormalLoadVector();
450
451 if ( nDim == 2 )
452 Pk *= elementProperties[0]; // thickness
453
454 boundaryEl.assembleIntoParentVector( Pk, fU );
455
456 break;
457 }
458 default: {
459 throw std::invalid_argument( "Invalid Load Type specified" );
460 }
461 }
462 }

464 template < int nDim, int nNodes >
465 void DisplacementFiniteElement< nDim, nNodes >::computeBodyForce( double* P_,
466 double* K,
467 const double* load,
468 const double* QTotal,
469 const double* time,
470 double dT )
471 {
472 Map< RhsSized > Pe( P_ );
473 const Map< const Matrix< double, nDim, 1 > > f( load );
474
475 for ( const auto& qp : qps )
476 Pe += this->NB( this->N( qp.xi ) ).transpose() * f * qp.J0xW;
477 }
