75 template < int n >
76 PerezFougetSubstepper< n >::PerezFougetSubstepper( double initialStepSize,
77 double minimumStepSize,
78 double scaleUpFactor,
79 double scaleDownFactor,
80 int nPassesToIncrease,
81 const Matrix6d& Cel )
82 :
83
84 initialStepSize( initialStepSize ),
85 minimumStepSize( minimumStepSize ),
86 scaleUpFactor( scaleUpFactor ),
87 scaleDownFactor( scaleDownFactor ),
88 nPassesToIncrease( nPassesToIncrease ),
89 currentProgress( 0.0 ),
90 currentSubstepSize( initialStepSize ),
91 passedSubsteps( 0 ),
92 Cel( Cel )
93 {
94 consistentTangent = MatrixStateStrain::Zero();
95 I76.setZero();
96 I76.topLeftCorner( 6, 6 ) = Matrix6d::Identity();
97 I77 = TangentSizedMatrix::Identity();
98 }
99
100 template < int n >
101 bool PerezFougetSubstepper< n >::isFinished()
102 {
103 // this is due to numerical accuracy ...
104 return ( 1.0 - currentProgress ) <= 2e-16;
105 }
106
107 template < int n >
108 double PerezFougetSubstepper< n >::getNextSubstep()
109 {
110 if ( passedSubsteps >= nPassesToIncrease )
111 currentSubstepSize *= scaleUpFactor;
112
113 const double remainingProgress = 1.0 - currentProgress;
114 if ( remainingProgress < currentSubstepSize )
115 currentSubstepSize = remainingProgress;
116
117 passedSubsteps++;
118 currentProgress += currentSubstepSize;
119
120 return currentSubstepSize;
121 }
122
123 template < int n >
124 bool PerezFougetSubstepper< n >::decreaseSubstepSize()
125 {
126 currentProgress -= currentSubstepSize;
127 passedSubsteps = 0;
128
129 currentSubstepSize *= scaleDownFactor;
130
131 if ( currentSubstepSize < minimumStepSize )
132 return MarmotJournal::warningToMSG( "UMAT: Substepper: Minimal stepzsize reached" );
133 else
134 return MarmotJournal::notificationToMSG( "UMAT: Substepper: Decreasing stepsize" );
135 }
136
137 template < int n >
138 void PerezFougetSubstepper< n >::finishElasticSubstep()
139 {
140 consistentTangent += currentSubstepSize * I76 * Cel;
141 }
142
143 template < int n >
144 void PerezFougetSubstepper< n >::finishSubstep( const TangentSizedMatrix& dXdY, const TangentSizedMatrix& dYdXOld )
145 {
146 consistentTangent.applyOnTheLeft( I77 - dYdXOld );
147 consistentTangent += currentSubstepSize * I76 * Cel;
148 consistentTangent.applyOnTheLeft( dXdY );
149 }
150
151 template < int n >
152 Matrix6d PerezFougetSubstepper< n >::consistentStiffness()
153 {
154 return consistentTangent.topLeftCorner( 6, 6 );
155 }
