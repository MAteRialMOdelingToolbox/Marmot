62 template < int sizeI, int sizeJ >
63 Eigen::Matrix< double, sizeI * sizeJ, sizeI * sizeJ > makeIndexSwapTensor()
64 {
65 // Aux. Matrix, which helps to swap indices in Eigen::Matrices abused as higher order Tensors by
66 // multiplication ,
67 //
68 // T_(ij)(kl) * IndexSwapTensor_(kl)(lk) = T_(ij)(lk)
69 //
70 // For instance:
71 //
72 // Matrix3d t;
73 // t<<1,2,3,4,5,6,7,8,9;
74 // const auto P = makeIndexSwapTensor<3,3>();
75 // std::cout << t.reshaped().transpose() << std::endl;
76 // std::cout << t.reshaped().transpose() * P << std::endl;
77 //
78 // Output:
79 //
80 // 1,4,7,2,5,8,3,6,9
81 // 1,2,3,4,5,6,7,8,9
82 //
83
84 Eigen::Matrix< double, sizeI * sizeJ, sizeI * sizeJ > P;
85
86 auto d = []( int a, int b ) -> double { return a == b ? 1.0 : 0.0; };
87
88 for ( int i = 0; i < sizeI; i++ )
89 for ( int j = 0; j < sizeJ; j++ )
90 for ( int l = 0; l < sizeI; l++ )
91 for ( int k = 0; k < sizeJ; k++ )
92 P( i + j * sizeI, l * sizeJ + k ) = d( i, l ) * d( k, j );
93
94 return P;
95 }

152 template < int nDim >
153 constexpr std::pair< int, int > fromVoigt( int ij )
154 {
155 if constexpr ( nDim == 1 )
156 return std::pair< int, int >( 0, 0 );
157 else if ( nDim == 2 )
158 switch ( ij ) {
159 case 0: return std::pair< int, int >( 0, 0 );
160 case 1: return std::pair< int, int >( 1, 1 );
161 case 2: return std::pair< int, int >( 0, 1 );
162 }
163
164 else if ( nDim == 3 ) {
165 switch ( ij ) {
166 case 0: return std::pair< int, int >( 0, 0 );
167 case 1: return std::pair< int, int >( 1, 1 );
168 case 2: return std::pair< int, int >( 2, 2 );
169 case 3: return std::pair< int, int >( 0, 1 );
170 case 4: return std::pair< int, int >( 0, 2 );
171 case 5: return std::pair< int, int >( 1, 2 );
172 }
173 }
174
175 throw std::invalid_argument( MakeString()
176 << __PRETTY_FUNCTION__ << ": invalid dimension / voigt index specified" );
177 }

179 template < int nDim >
180 constexpr int toVoigt( int i, int j )
181 {
182 if constexpr ( nDim == 1 )
183 return 0;
184 else if ( nDim == 2 )
185 return ( i == j ) ? ( i == 0 ? 0 : 1 ) : 2;
186
187 else if ( nDim == 3 ) {
188 constexpr int tensor2VoigtNotationIndicesMapping[3][3] = { { 0, 3, 4 }, { 3, 1, 5 }, { 4, 5, 2 } };
189 return tensor2VoigtNotationIndicesMapping[i][j];
190 }
191
192 throw std::invalid_argument( MakeString() << __PRETTY_FUNCTION__ << ": invalid dimension specified" );
193 }

195 template < int nDim >
196 Eigen::TensorFixedSize< double, Eigen::Sizes< VOIGTFROMDIM( nDim ), nDim, nDim > > voigtMap()
197 {
198 using namespace Eigen;
199 Eigen::TensorFixedSize< double, Eigen::Sizes< VOIGTFROMDIM( nDim ), nDim, nDim > > result;
200 result.setZero();
201 for ( int i = 0; i < nDim; i++ )
202 for ( int j = 0; j < nDim; j++ )
203 result( toVoigt< nDim >( i, j ), i, j ) = 1;
204 return result;
205 }
